100      /**
101       * Find the original TGT within the current subject's credentials. Cross-realm
102       * TGT's of the form "krbtgt/TWO.COM@ONE.COM" may be present.
103       * 
104       * @return The TGT from the current subject
105       * @throws IOException
106       *           if TGT can't be found
107       */
108      private static KerberosTicket getTgtFromSubject() throws IOException {
109        Subject current = Subject.getSubject(AccessController.getContext());
110        if (current == null) {
111          throw new IOException(
112              "Can't get TGT from current Subject, because it is null");
113        }
114        Set<KerberosTicket> tickets = current
115            .getPrivateCredentials(KerberosTicket.class);
116        for (KerberosTicket t : tickets) {
117          if (isOriginalTGT(t))
118            return t;
119        }
120        throw new IOException("Failed to find TGT from current Subject:"+current);
121      }

123      /**
124       * TGS must have the server principal of the form "krbtgt/FOO@FOO".
125       * @param principal
126       * @return true or false
127       */
128      static boolean 
129      isTGSPrincipal(KerberosPrincipal principal) {
130        if (principal == null)
131          return false;
132        if (principal.getName().equals("krbtgt/" + principal.getRealm() + 
133            "@" + principal.getRealm())) {
134          return true;
135        }
136        return false;
137      }

149      /**
150       * Convert Kerberos principal name pattern to valid Kerberos principal
151       * names. It replaces hostname pattern with hostname, which should be
152       * fully-qualified domain name. If hostname is null or "0.0.0.0", it uses
153       * dynamically looked-up fqdn of the current host instead.
154       * 
155       * @param principalConfig
156       *          the Kerberos principal name conf value to convert
157       * @param hostname
158       *          the fully-qualified domain name used for substitution
159       * @return converted Kerberos principal name
160       * @throws IOException if the client address cannot be determined
161       */
162      @InterfaceAudience.Public
163      @InterfaceStability.Evolving
164      public static String getServerPrincipal(String principalConfig,
165          String hostname) throws IOException {
166        String[] components = getComponents(principalConfig);
167        if (components == null || components.length != 3
168            || !components[1].equals(HOSTNAME_PATTERN)) {
169          return principalConfig;
170        } else {
171          return replacePattern(components, hostname);
172        }
173      }
174      
175      /**
176       * Convert Kerberos principal name pattern to valid Kerberos principal names.
177       * This method is similar to {@link #getServerPrincipal(String, String)},
178       * except 1) the reverse DNS lookup from addr to hostname is done only when
179       * necessary, 2) param addr can't be null (no default behavior of using local
180       * hostname when addr is null).
181       * 
182       * @param principalConfig
183       *          Kerberos principal name pattern to convert
184       * @param addr
185       *          InetAddress of the host used for substitution
186       * @return converted Kerberos principal name
187       * @throws IOException if the client address cannot be determined
188       */
189      @InterfaceAudience.Public
190      @InterfaceStability.Evolving
191      public static String getServerPrincipal(String principalConfig,
192          InetAddress addr) throws IOException {
193        String[] components = getComponents(principalConfig);
194        if (components == null || components.length != 3
195            || !components[1].equals(HOSTNAME_PATTERN)) {
196          return principalConfig;
197        } else {
198          if (addr == null) {
199            throw new IOException("Can't replace " + HOSTNAME_PATTERN
200                + " pattern since client address is null");
201          }
202          return replacePattern(components, addr.getCanonicalHostName());
203        }
204      }
205      
206      private static String[] getComponents(String principalConfig) {
207        if (principalConfig == null)
208          return null;
209        return principalConfig.split("[/@]");
210      }
211      
212      private static String replacePattern(String[] components, String hostname)
213          throws IOException {
214        String fqdn = hostname;
215        if (fqdn == null || fqdn.equals("") || fqdn.equals("0.0.0.0")) {
216          fqdn = getLocalHostName();
217        }
218        return components[0] + "/" + fqdn.toLowerCase() + "@" + components[2];
219      }
220      
221      static String getLocalHostName() throws UnknownHostException {
222        return InetAddress.getLocalHost().getCanonicalHostName();
223      }

225      /**
226       * Login as a principal specified in config. Substitute $host in
227       * user's Kerberos principal name with a dynamically looked-up fully-qualified
228       * domain name of the current host.
229       * 
230       * @param conf
231       *          conf to use
232       * @param keytabFileKey
233       *          the key to look for keytab file in conf
234       * @param userNameKey
235       *          the key to look for user's Kerberos principal name in conf
236       * @throws IOException if login fails
237       */
238      @InterfaceAudience.Public
239      @InterfaceStability.Evolving
240      public static void login(final Configuration conf,
241          final String keytabFileKey, final String userNameKey) throws IOException {
242        login(conf, keytabFileKey, userNameKey, getLocalHostName());
243      }

245      /**
246       * Login as a principal specified in config. Substitute $host in user's Kerberos principal 
247       * name with hostname. If non-secure mode - return. If no keytab available -
248       * bail out with an exception
249       * 
250       * @param conf
251       *          conf to use
252       * @param keytabFileKey
253       *          the key to look for keytab file in conf
254       * @param userNameKey
255       *          the key to look for user's Kerberos principal name in conf
256       * @param hostname
257       *          hostname to use for substitution
258       * @throws IOException if the config doesn't specify a keytab
259       */
260      @InterfaceAudience.Public
261      @InterfaceStability.Evolving
262      public static void login(final Configuration conf,
263          final String keytabFileKey, final String userNameKey, String hostname)
264          throws IOException {
265        
266        if(! UserGroupInformation.isSecurityEnabled()) 
267          return;
268        
269        String keytabFilename = conf.get(keytabFileKey);
270        if (keytabFilename == null || keytabFilename.length() == 0) {
271          throw new IOException("Running in secure mode, but config doesn't have a keytab");
272        }
273    
274        String principalConfig = conf.get(userNameKey, System
275            .getProperty("user.name"));
276        String principalName = SecurityUtil.getServerPrincipal(principalConfig,
277            hostname);
278        UserGroupInformation.loginUserFromKeytab(principalName, keytabFilename);
279      }

283       * @param uri of the service
284       * @param defPort is used if the uri lacks a port
285       * @return the token service, or null if no authority
286       * @see #buildTokenService(InetSocketAddress)

288      public static String buildDTServiceName(URI uri, int defPort) {
289        String authority = uri.getAuthority();
290        if (authority == null) {

292        }
293        InetSocketAddress addr = NetUtils.createSocketAddr(authority, defPort);
294        return buildTokenService(addr).toString();

323       * @param conf configuration object
324       * @return the KerberosInfo or null if it has no KerberosInfo defined

331            if (result != null) {

336        
337        synchronized (securityInfoProviders) {

340            if (result != null) {
341              return result;
342            }
343          }
344        }
345        return null;
346      }

403      /**
404       * Construct the service key for a token
405       * @param addr InetSocketAddress of remote connection with a token
406       * @return "ip:port" or "host:port" depending on the value of
407       *          hadoop.security.token.service.use_ip
408       */
409      public static Text buildTokenService(InetSocketAddress addr) {
410        String host = null;
411        if (useIpForTokenService) {
412          if (addr.isUnresolved()) { // host has no ip address
413            throw new IllegalArgumentException(
414                new UnknownHostException(addr.getHostName())
415            );
416          }
417          host = addr.getAddress().getHostAddress();
418        } else {
419          host = addr.getHostName().toLowerCase();
420        }
421        return new Text(host + ":" + addr.getPort());
422      }

434      /**
435       * Perform the given action as the daemon's login user. If the login
436       * user cannot be determined, this will log a FATAL error and exit
437       * the whole JVM.
438       */
439      public static <T> T doAsLoginUserOrFatal(PrivilegedAction<T> action) { 
440        if (UserGroupInformation.isSecurityEnabled()) {
441          UserGroupInformation ugi = null;
442          try { 
443            ugi = UserGroupInformation.getLoginUser();
444          } catch (IOException e) {
445            LOG.fatal("Exception while getting login user", e);
446            e.printStackTrace();
447            Runtime.getRuntime().exit(-1);
448          }
449          return ugi.doAs(action);
450        } else {
451          return action.run();
452        }
453      }

456       * Open a (if need be) secure connection to a URL in a secure environment
457       * that is using SPNEGO to authenticate its URLs. All Namenode and Secondary
458       * Namenode URLs that are protected via SPNEGO should be accessed via this
459       * method.
460       *
461       * @param url to authenticate via SPNEGO.
462       * @return A connection that has been authenticated via SPNEGO
463       * @throws IOException If unable to authenticate via SPNEGO
464       */
465      public static URLConnection openSecureHttpConnection(URL url) throws IOException {
466        if(!UserGroupInformation.isSecurityEnabled()) {
467          return url.openConnection();
468        }
469    
470        AuthenticatedURL.Token token = new AuthenticatedURL.Token();
471        try {
472          return new AuthenticatedURL(null, sslFactory).openConnection(url, token);
473        } catch (AuthenticationException e) {
474          throw new IOException("Exception trying to open authenticated connection to "
475                  + url, e);
476        }
477      }

507      /**
508       * This an alternate resolver with important properties that the standard
509       * java resolver lacks:
510       * 1) The hostname is fully qualified.  This avoids security issues if not
511       *    all hosts in the cluster do not share the same search domains.  It
512       *    also prevents other hosts from performing unnecessary dns searches.
513       *    In contrast, InetAddress simply returns the host as given.
514       * 2) The InetAddress is instantiated with an exact host and IP to prevent
515       *    further unnecessary lookups.  InetAddress may perform an unnecessary
516       *    reverse lookup for an IP.
517       * 3) A call to getHostName() will always return the qualified hostname, or
518       *    more importantly, the IP if instantiated with an IP.  This avoids
519       *    unnecessary dns timeouts if the host is not resolvable.
520       * 4) Point 3 also ensures that if the host is re-resolved, ex. during a
521       *    connection re-attempt, that a reverse lookup to host and forward
522       *    lookup to IP is not performed since the reverse/forward mappings may
523       *    not always return the same IP.  If the client initiated a connection
524       *    with an IP, then that IP is all that should ever be contacted.
525       *    
526       * NOTE: this resolver is only used if:
527       *       hadoop.security.token.service.use_ip=false 
528       */
529      protected static class QualifiedHostResolver implements HostResolver {
530        @SuppressWarnings("unchecked")
531        private List<String> searchDomains =
532            ResolverConfiguration.open().searchlist();
533        
534        /**
535         * Create an InetAddress with a fully qualified hostname of the given
536         * hostname.  InetAddress does not qualify an incomplete hostname that
537         * is resolved via the domain search list.
538         * {@link InetAddress#getCanonicalHostName()} will fully qualify the
539         * hostname, but it always return the A record whereas the given hostname
540         * may be a CNAME.
541         * 
542         * @param host a hostname or ip address
543         * @return InetAddress with the fully qualified hostname or ip
544         * @throws UnknownHostException if host does not exist
545         */
546        @Override
547        public InetAddress getByName(String host) throws UnknownHostException {
548          InetAddress addr = null;
549    
550          if (IPAddressUtil.isIPv4LiteralAddress(host)) {
551            // use ipv4 address as-is
552            byte[] ip = IPAddressUtil.textToNumericFormatV4(host);
553            addr = InetAddress.getByAddress(host, ip);
554          } else if (IPAddressUtil.isIPv6LiteralAddress(host)) {
555            // use ipv6 address as-is
556            byte[] ip = IPAddressUtil.textToNumericFormatV6(host);
557            addr = InetAddress.getByAddress(host, ip);
558          } else if (host.endsWith(".")) {
559            // a rooted host ends with a dot, ex. "host."
560            // rooted hosts never use the search path, so only try an exact lookup
561            addr = getByExactName(host);
562          } else if (host.contains(".")) {
563            // the host contains a dot (domain), ex. "host.domain"
564            // try an exact host lookup, then fallback to search list
565            addr = getByExactName(host);
566            if (addr == null) {
567              addr = getByNameWithSearch(host);
568            }
569          } else {
570            // it's a simple host with no dots, ex. "host"
571            // try the search list, then fallback to exact host
572            InetAddress loopback = InetAddress.getByName(null);
573            if (host.equalsIgnoreCase(loopback.getHostName())) {
574              addr = InetAddress.getByAddress(host, loopback.getAddress());
575            } else {
576              addr = getByNameWithSearch(host);
577              if (addr == null) {
578                addr = getByExactName(host);
579              }
580            }
581          }
582          // unresolvable!
583          if (addr == null) {
584            throw new UnknownHostException(host);
585          }
586          return addr;
587        }
588    
589        InetAddress getByExactName(String host) {
590          InetAddress addr = null;
591          // InetAddress will use the search list unless the host is rooted
592          // with a trailing dot.  The trailing dot will disable any use of the
593          // search path in a lower level resolver.  See RFC 1535.
594          String fqHost = host;
595          if (!fqHost.endsWith(".")) fqHost += ".";
596          try {
597            addr = getInetAddressByName(fqHost);
598            // can't leave the hostname as rooted or other parts of the system
599            // malfunction, ex. kerberos principals are lacking proper host
600            // equivalence for rooted/non-rooted hostnames
601            addr = InetAddress.getByAddress(host, addr.getAddress());
602          } catch (UnknownHostException e) {
603            // ignore, caller will throw if necessary
604          }
605          return addr;
606        }
607    
608        InetAddress getByNameWithSearch(String host) {
609          InetAddress addr = null;
610          if (host.endsWith(".")) { // already qualified?
611            addr = getByExactName(host); 
612          } else {
613            for (String domain : searchDomains) {
614              String dot = !domain.startsWith(".") ? "." : "";
615              addr = getByExactName(host + dot + domain);
616              if (addr != null) break;
617            }
618          }
619          return addr;
620        }
621    
622        // implemented as a separate method to facilitate unit testing
623        InetAddress getInetAddressByName(String host) throws UnknownHostException {
624          return InetAddress.getByName(host);
625        }
626    
627        void setSearchDomains(String ... domains) {
628          searchDomains = Arrays.asList(domains);
629        }
630      }
