018    package org.apache.hadoop.hdfs.server.namenode;
019    
020    import java.io.Closeable;
021    import java.io.IOException;
022    import java.util.Collection;
023    
024    import org.apache.hadoop.classification.InterfaceAudience;
025    import org.apache.hadoop.classification.InterfaceStability;
026    
027    /**
028     * A JournalManager is responsible for managing a single place of storing
029     * edit logs. It may correspond to multiple files, a backup node, etc.
030     * Even when the actual underlying storage is rolled, or failed and restored,
031     * each conceptual place of storage corresponds to exactly one instance of
032     * this class, which is created when the EditLog is first opened.
033     */
034    @InterfaceAudience.Private
035    @InterfaceStability.Evolving
036    public interface JournalManager extends Closeable {
037      /**
038       * Begin writing to a new segment of the log stream, which starts at
039       * the given transaction ID.
040       */
041      EditLogOutputStream startLogSegment(long txId) throws IOException;
042    
043      /**
044       * Mark the log segment that spans from firstTxId to lastTxId
045       * as finalized and complete.
046       */
047      void finalizeLogSegment(long firstTxId, long lastTxId) throws IOException;
048    
049       /**
050       * Get a list of edit log input streams.  The list will start with the
051       * stream that contains fromTxnId, and continue until the end of the journal
052       * being managed.
053       * 
054       * @param fromTxnId the first transaction id we want to read
055       * @param inProgressOk whether or not in-progress streams should be returned
056       *
057       * @return a list of streams
058       */
059      void selectInputStreams(Collection<EditLogInputStream> streams,
060          long fromTxnId, boolean inProgressOk);
061    
062      /**
063       * Set the amount of memory that this stream should use to buffer edits
064       */
065      void setOutputBufferCapacity(int size);
066    
067      /**
068       * The JournalManager may archive/purge any logs for transactions less than
069       * or equal to minImageTxId.
070       *
071       * @param minTxIdToKeep the earliest txid that must be retained after purging
072       *                      old logs
073       * @throws IOException if purging fails
074       */
075      void purgeLogsOlderThan(long minTxIdToKeep)
076        throws IOException;
077    
078      /**
079       * Recover segments which have not been finalized.
080       */
081      void recoverUnfinalizedSegments() throws IOException;
082    
083      /**
084       * Close the journal manager, freeing any resources it may hold.
085       */
086      @Override
087      void close() throws IOException;
088      
089      /** 
090       * Indicate that a journal is cannot be used to load a certain range of 
091       * edits.
092       * This exception occurs in the case of a gap in the transactions, or a
093       * corrupt edit file.
094       */
095      public static class CorruptionException extends IOException {
096        static final long serialVersionUID = -4687802717006172702L;
097        
098        public CorruptionException(String reason) {
099          super(reason);
100        }
101      }
102    }
