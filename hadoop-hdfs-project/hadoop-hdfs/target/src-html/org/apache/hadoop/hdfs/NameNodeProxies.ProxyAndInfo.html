073    /**
074     * Create proxy objects to communicate with a remote NN. All remote access to an
075     * NN should be funneled through this class. Most of the time you'll want to use
076     * {@link NameNodeProxies#createProxy(Configuration, URI, Class)}, which will
077     * create either an HA- or non-HA-enabled client proxy as appropriate.
078     */
079    public class NameNodeProxies {
080      
081      private static final Log LOG = LogFactory.getLog(NameNodeProxies.class);
082    
083      /**
084       * Wrapper for a client proxy as well as its associated service ID.
085       * This is simply used as a tuple-like return type for
086       * {@link NameNodeProxies#createProxy} and
087       * {@link NameNodeProxies#createNonHAProxy}.
088       */
089      public static class ProxyAndInfo<PROXYTYPE> {
090        private final PROXYTYPE proxy;
091        private final Text dtService;
092        
093        public ProxyAndInfo(PROXYTYPE proxy, Text dtService) {
094          this.proxy = proxy;
095          this.dtService = dtService;
096        }
097        
098        public PROXYTYPE getProxy() {
099          return proxy;
100        }
101        
102        public Text getDelegationTokenService() {
103          return dtService;
104        }
105      }
106    
107      /**
108       * Creates the namenode proxy with the passed protocol. This will handle
109       * creation of either HA- or non-HA-enabled proxy objects, depending upon
110       * if the provided URI is a configured logical URI.
111       * 
112       * @param conf the configuration containing the required IPC
113       *        properties, client failover configurations, etc.
114       * @param nameNodeUri the URI pointing either to a specific NameNode
115       *        or to a logical nameservice.
116       * @param xface the IPC interface which should be created
117       * @return an object containing both the proxy and the associated
118       *         delegation token service it corresponds to
119       * @throws IOException if there is an error creating the proxy
120       **/
121      @SuppressWarnings("unchecked")
122      public static <T> ProxyAndInfo<T> createProxy(Configuration conf,
123          URI nameNodeUri, Class<T> xface) throws IOException {
124        Class<FailoverProxyProvider<T>> failoverProxyProviderClass =
125            getFailoverProxyProviderClass(conf, nameNodeUri, xface);
126      
127        if (failoverProxyProviderClass == null) {
128          // Non-HA case
129          return createNonHAProxy(conf, NameNode.getAddress(nameNodeUri), xface,
130              UserGroupInformation.getCurrentUser(), true);
131        } else {
132          // HA case
133          FailoverProxyProvider<T> failoverProxyProvider = NameNodeProxies
134              .createFailoverProxyProvider(conf, failoverProxyProviderClass, xface,
135                  nameNodeUri);
136          Conf config = new Conf(conf);
137          T proxy = (T) RetryProxy.create(xface, failoverProxyProvider, RetryPolicies
138              .failoverOnNetworkException(RetryPolicies.TRY_ONCE_THEN_FAIL,
139                  config.maxFailoverAttempts, config.failoverSleepBaseMillis,
140                  config.failoverSleepMaxMillis));
141          
142          Text dtService = HAUtil.buildTokenServiceForLogicalUri(nameNodeUri);
143          return new ProxyAndInfo<T>(proxy, dtService);
144        }
145      }

147      /**
148       * Creates an explicitly non-HA-enabled proxy object. Most of the time you
149       * don't want to use this, and should instead use {@link NameNodeProxies#createProxy}.
150       * 
151       * @param conf the configuration object
152       * @param nnAddr address of the remote NN to connect to
153       * @param xface the IPC interface which should be created
154       * @param ugi the user who is making the calls on the proxy object
155       * @param withRetries certain interfaces have a non-standard retry policy
156       * @return an object containing both the proxy and the associated
157       *         delegation token service it corresponds to
158       * @throws IOException
159       */
160      @SuppressWarnings("unchecked")
161      public static <T> ProxyAndInfo<T> createNonHAProxy(
162          Configuration conf, InetSocketAddress nnAddr, Class<T> xface,
163          UserGroupInformation ugi, boolean withRetries) throws IOException {
164        Text dtService = SecurityUtil.buildTokenService(nnAddr);
165      
166        T proxy;
167        if (xface == ClientProtocol.class) {
168          proxy = (T) createNNProxyWithClientProtocol(nnAddr, conf, ugi,
169              withRetries);
170        } else if (xface == JournalProtocol.class) {
171          proxy = (T) createNNProxyWithJournalProtocol(nnAddr, conf, ugi);
172        } else if (xface == NamenodeProtocol.class) {
173          proxy = (T) createNNProxyWithNamenodeProtocol(nnAddr, conf, ugi,
174              withRetries);
175        } else if (xface == GetUserMappingsProtocol.class) {
176          proxy = (T) createNNProxyWithGetUserMappingsProtocol(nnAddr, conf, ugi);
177        } else if (xface == RefreshUserMappingsProtocol.class) {
178          proxy = (T) createNNProxyWithRefreshUserMappingsProtocol(nnAddr, conf, ugi);
179        } else if (xface == RefreshAuthorizationPolicyProtocol.class) {
180          proxy = (T) createNNProxyWithRefreshAuthorizationPolicyProtocol(nnAddr,
181              conf, ugi);
182        } else {
183          String message = "Upsupported protocol found when creating the proxy " +
184              "connection to NameNode: " +
185              ((xface != null) ? xface.getClass().getName() : "null");
186          LOG.error(message);
187          throw new IllegalStateException(message);
188        }
189        return new ProxyAndInfo<T>(proxy, dtService);
190      }

224      private static NamenodeProtocol createNNProxyWithNamenodeProtocol(
225          InetSocketAddress address, Configuration conf, UserGroupInformation ugi,
226          boolean withRetries) throws IOException {
227        NamenodeProtocolPB proxy = (NamenodeProtocolPB) createNameNodeProxy(
228            address, conf, ugi, NamenodeProtocolPB.class);
229        if (withRetries) { // create the proxy with retries
230          RetryPolicy timeoutPolicy = RetryPolicies.exponentialBackoffRetry(5, 200,
231              TimeUnit.MILLISECONDS);
232          Map<Class<? extends Exception>, RetryPolicy> exceptionToPolicyMap 
233                         = new HashMap<Class<? extends Exception>, RetryPolicy>();
234          RetryPolicy methodPolicy = RetryPolicies.retryByException(timeoutPolicy,
235              exceptionToPolicyMap);
236          Map<String, RetryPolicy> methodNameToPolicyMap 
237                         = new HashMap<String, RetryPolicy>();
238          methodNameToPolicyMap.put("getBlocks", methodPolicy);
239          methodNameToPolicyMap.put("getAccessKeys", methodPolicy);
240          proxy = (NamenodeProtocolPB) RetryProxy.create(NamenodeProtocolPB.class,
241              proxy, methodNameToPolicyMap);
242        }
243        return new NamenodeProtocolTranslatorPB(proxy);
244      }

246      /**
247       * Return the default retry policy used in RPC.
248       * 
249       * If dfs.client.retry.policy.enabled == false, use TRY_ONCE_THEN_FAIL.
250       * 
251       * Otherwise, first unwrap ServiceException if possible, and then 
252       * (1) use multipleLinearRandomRetry for
253       *     - SafeModeException, or
254       *     - IOException other than RemoteException, or
255       *     - ServiceException; and
256       * (2) use TRY_ONCE_THEN_FAIL for
257       *     - non-SafeMode RemoteException, or
258       *     - non-IOException.
259       *     
260       * Note that dfs.client.retry.max < 0 is not allowed.
261       */
262      public static RetryPolicy getDefaultRetryPolicy(Configuration conf) {
263        final RetryPolicy multipleLinearRandomRetry = getMultipleLinearRandomRetry(conf);
264        if (LOG.isDebugEnabled()) {
265          LOG.debug("multipleLinearRandomRetry = " + multipleLinearRandomRetry);
266        }
267        if (multipleLinearRandomRetry == null) {
268          //no retry
269          return RetryPolicies.TRY_ONCE_THEN_FAIL;
270        } else {
271          return new RetryPolicy() {
272            @Override
273            public RetryAction shouldRetry(Exception e, int retries, int failovers,
274                boolean isMethodIdempotent) throws Exception {
275              if (e instanceof ServiceException) {
276                //unwrap ServiceException
277                final Throwable cause = e.getCause();
278                if (cause != null && cause instanceof Exception) {
279                  e = (Exception)cause;
280                }
281              }
282    
283              //see (1) and (2) in the javadoc of this method.
284              final RetryPolicy p;
285              if (e instanceof RemoteException) {
286                final RemoteException re = (RemoteException)e;
287                p = SafeModeException.class.getName().equals(re.getClassName())?
288                    multipleLinearRandomRetry: RetryPolicies.TRY_ONCE_THEN_FAIL;
289              } else if (e instanceof IOException || e instanceof ServiceException) {
290                p = multipleLinearRandomRetry;
291              } else { //non-IOException
292                p = RetryPolicies.TRY_ONCE_THEN_FAIL;
293              }
294    
295              if (LOG.isDebugEnabled()) {
296                LOG.debug("RETRY " + retries + ") policy="
297                    + p.getClass().getSimpleName() + ", exception=" + e);
298              }
299              LOG.info("RETRY " + retries + ") policy="
300                  + p.getClass().getSimpleName() + ", exception=" + e);
301              return p.shouldRetry(e, retries, failovers, isMethodIdempotent);
302            }
303    
304            @Override
305            public String toString() {
306              return "RetryPolicy[" + multipleLinearRandomRetry + ", "
307                  + RetryPolicies.TRY_ONCE_THEN_FAIL.getClass().getSimpleName()
308                  + "]";
309            }
310          };
311        }
312      }

314      /**
315       * Return the MultipleLinearRandomRetry policy specified in the conf,
316       * or null if the feature is disabled.
317       * If the policy is specified in the conf but the policy cannot be parsed,
318       * the default policy is returned.
319       * 
320       * Conf property: N pairs of sleep-time and number-of-retries
321       *   dfs.client.retry.policy = "s1,n1,s2,n2,..."
322       */
323      private static RetryPolicy getMultipleLinearRandomRetry(Configuration conf) {
324        final boolean enabled = conf.getBoolean(
325            DFSConfigKeys.DFS_CLIENT_RETRY_POLICY_ENABLED_KEY,
326            DFSConfigKeys.DFS_CLIENT_RETRY_POLICY_ENABLED_DEFAULT);
327        if (!enabled) {
328          return null;
329        }
330    
331        final String policy = conf.get(
332            DFSConfigKeys.DFS_CLIENT_RETRY_POLICY_SPEC_KEY,
333            DFSConfigKeys.DFS_CLIENT_RETRY_POLICY_SPEC_DEFAULT);
334    
335        final RetryPolicy r = RetryPolicies.MultipleLinearRandomRetry.parseCommaSeparatedString(policy);
336        return r != null? r: RetryPolicies.MultipleLinearRandomRetry.parseCommaSeparatedString(
337            DFSConfigKeys.DFS_CLIENT_RETRY_POLICY_SPEC_DEFAULT);
338      }

340      private static ClientProtocol createNNProxyWithClientProtocol(
341          InetSocketAddress address, Configuration conf, UserGroupInformation ugi,
342          boolean withRetries) throws IOException {
343        RPC.setProtocolEngine(conf, ClientNamenodeProtocolPB.class, ProtobufRpcEngine.class);
344    
345        final RetryPolicy defaultPolicy = getDefaultRetryPolicy(conf);
346        final long version = RPC.getProtocolVersion(ClientNamenodeProtocolPB.class);
347        ClientNamenodeProtocolPB proxy = RPC.getProtocolProxy(
348            ClientNamenodeProtocolPB.class, version, address, ugi, conf,
349            NetUtils.getDefaultSocketFactory(conf), 0, defaultPolicy).getProxy();
350    
351        if (withRetries) { // create the proxy with retries
352    
353          RetryPolicy createPolicy = RetryPolicies
354              .retryUpToMaximumCountWithFixedSleep(5,
355                  HdfsConstants.LEASE_SOFTLIMIT_PERIOD, TimeUnit.MILLISECONDS);
356        
357          Map<Class<? extends Exception>, RetryPolicy> remoteExceptionToPolicyMap 
358                     = new HashMap<Class<? extends Exception>, RetryPolicy>();
359          remoteExceptionToPolicyMap.put(AlreadyBeingCreatedException.class,
360              createPolicy);
361        
362          Map<Class<? extends Exception>, RetryPolicy> exceptionToPolicyMap
363                     = new HashMap<Class<? extends Exception>, RetryPolicy>();
364          exceptionToPolicyMap.put(RemoteException.class, RetryPolicies
365              .retryByRemoteException(defaultPolicy,
366                  remoteExceptionToPolicyMap));
367          RetryPolicy methodPolicy = RetryPolicies.retryByException(
368              defaultPolicy, exceptionToPolicyMap);
369          Map<String, RetryPolicy> methodNameToPolicyMap 
370                     = new HashMap<String, RetryPolicy>();
371        
372          methodNameToPolicyMap.put("create", methodPolicy);
373        
374          proxy = (ClientNamenodeProtocolPB) RetryProxy.create(
375              ClientNamenodeProtocolPB.class,
376              new DefaultFailoverProxyProvider<ClientNamenodeProtocolPB>(
377                  ClientNamenodeProtocolPB.class, proxy),
378              methodNameToPolicyMap,
379              defaultPolicy);
380        }
381        return new ClientNamenodeProtocolTranslatorPB(proxy);
382      }

384      @SuppressWarnings("unchecked")
385      private static Object createNameNodeProxy(InetSocketAddress address,
386          Configuration conf, UserGroupInformation ugi, Class xface)
387          throws IOException {
388        RPC.setProtocolEngine(conf, xface, ProtobufRpcEngine.class);
389        Object proxy = RPC.getProxy(xface, RPC.getProtocolVersion(xface), address,
390            ugi, conf, NetUtils.getDefaultSocketFactory(conf));
391        return proxy;
392      }

394      /** Gets the configured Failover proxy provider's class */
395      private static <T> Class<FailoverProxyProvider<T>> getFailoverProxyProviderClass(
396          Configuration conf, URI nameNodeUri, Class<T> xface) throws IOException {
397        if (nameNodeUri == null) {
398          return null;
399        }
400        String host = nameNodeUri.getHost();
401      
402        String configKey = DFS_CLIENT_FAILOVER_PROXY_PROVIDER_KEY_PREFIX + "."
403            + host;
404        try {
405          @SuppressWarnings("unchecked")
406          Class<FailoverProxyProvider<T>> ret = (Class<FailoverProxyProvider<T>>) conf
407              .getClass(configKey, null, FailoverProxyProvider.class);
408          if (ret != null) {
409            // If we found a proxy provider, then this URI should be a logical NN.
410            // Given that, it shouldn't have a non-default port number.
411            int port = nameNodeUri.getPort();
412            if (port > 0 && port != NameNode.DEFAULT_PORT) {
413              throw new IOException("Port " + port + " specified in URI "
414                  + nameNodeUri + " but host '" + host
415                  + "' is a logical (HA) namenode"
416                  + " and does not use port information.");
417            }
418          }
419          return ret;
420        } catch (RuntimeException e) {
421          if (e.getCause() instanceof ClassNotFoundException) {
422            throw new IOException("Could not load failover proxy provider class "
423                + conf.get(configKey) + " which is configured for authority "
424                + nameNodeUri, e);
425          } else {
426            throw e;
427          }
428        }
429      }

431      /** Creates the Failover proxy provider instance*/
432      @SuppressWarnings("unchecked")
433      private static <T> FailoverProxyProvider<T> createFailoverProxyProvider(
434          Configuration conf, Class<FailoverProxyProvider<T>> failoverProxyProviderClass,
435          Class<T> xface, URI nameNodeUri) throws IOException {
436        Preconditions.checkArgument(
437            xface.isAssignableFrom(NamenodeProtocols.class),
438            "Interface %s is not a NameNode protocol", xface);
439        try {
440          Constructor<FailoverProxyProvider<T>> ctor = failoverProxyProviderClass
441              .getConstructor(Configuration.class, URI.class, Class.class);
442          FailoverProxyProvider<?> provider = ctor.newInstance(conf, nameNodeUri,
443              xface);
444          return (FailoverProxyProvider<T>) provider;
445        } catch (Exception e) {
446          String message = "Couldn't create proxy provider " + failoverProxyProviderClass;
447          if (LOG.isDebugEnabled()) {
448            LOG.debug(message, e);
449          }
450          if (e.getCause() instanceof IOException) {
451            throw (IOException) e.getCause();
452          } else {
453            throw new IOException(message, e);
454          }
455        }
456      }
