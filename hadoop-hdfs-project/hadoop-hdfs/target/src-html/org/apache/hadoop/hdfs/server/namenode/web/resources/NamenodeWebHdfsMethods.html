143      static DatanodeInfo chooseDatanode(final NameNode namenode,
144          final String path, final HttpOpParam.Op op, final long openOffset,
145          final long blocksize, Configuration conf) throws IOException {
146        final BlockManager bm = namenode.getNamesystem().getBlockManager();
147    
148        if (op == PutOpParam.Op.CREATE) {
149          //choose a datanode near to client 
150          final DatanodeDescriptor clientNode = bm.getDatanodeManager(
151              ).getDatanodeByHost(getRemoteAddress());
152          if (clientNode != null) {
153            final DatanodeDescriptor[] datanodes = bm.getBlockPlacementPolicy(
154                ).chooseTarget(path, 1, clientNode, null, blocksize);
155            if (datanodes.length > 0) {
156              return datanodes[0];
157            }
158          }
159        } else if (op == GetOpParam.Op.OPEN
160            || op == GetOpParam.Op.GETFILECHECKSUM
161            || op == PostOpParam.Op.APPEND) {
162          //choose a datanode containing a replica 
163          final NamenodeProtocols np = namenode.getRpcServer();
164          final HdfsFileStatus status = np.getFileInfo(path);
165          if (status == null) {
166            throw new FileNotFoundException("File " + path + " not found.");
167          }
168          final long len = status.getLen();
169          if (op == GetOpParam.Op.OPEN) {
170            if (openOffset < 0L || (openOffset >= len && len > 0)) {
171              throw new IOException("Offset=" + openOffset
172                  + " out of the range [0, " + len + "); " + op + ", path=" + path);
173            }
174          }
175    
176          if (len > 0) {
177            final long offset = op == GetOpParam.Op.OPEN? openOffset: len - 1;
178            final LocatedBlocks locations = np.getBlockLocations(path, offset, 1);
179            final int count = locations.locatedBlockCount();
180            if (count > 0) {
181              return JspHelper.bestNode(locations.get(0).getLocations(), false, conf);
182            }
183          }
184        } 
185    
186        return (DatanodeDescriptor)bm.getDatanodeManager().getNetworkTopology(
187            ).chooseRandom(NodeBase.ROOT);
188      }

190      private Token<? extends TokenIdentifier> generateDelegationToken(
191          final NameNode namenode, final UserGroupInformation ugi,
192          final String renewer) throws IOException {
193        final Credentials c = DelegationTokenSecretManager.createCredentials(
194            namenode, ugi, renewer != null? renewer: ugi.getShortUserName());
195        final Token<? extends TokenIdentifier> t = c.getAllTokens().iterator().next();
196        t.setKind(WebHdfsFileSystem.TOKEN_KIND);
197        SecurityUtil.setTokenService(t, namenode.getHttpAddress());
198        return t;
199      }

201      private URI redirectURI(final NameNode namenode,
202          final UserGroupInformation ugi, final DelegationParam delegation,
203          final UserParam username, final DoAsParam doAsUser,
204          final String path, final HttpOpParam.Op op, final long openOffset,
205          final long blocksize,
206          final Param<?, ?>... parameters) throws URISyntaxException, IOException {
207        final Configuration conf = (Configuration)context.getAttribute(JspHelper.CURRENT_CONF);
208        final DatanodeInfo dn = chooseDatanode(namenode, path, op, openOffset,
209            blocksize, conf);
210    
211        final String delegationQuery;
212        if (!UserGroupInformation.isSecurityEnabled()) {
213          //security disabled
214          delegationQuery = Param.toSortedString("&", doAsUser, username);
215        } else if (delegation.getValue() != null) {
216          //client has provided a token
217          delegationQuery = "&" + delegation;
218        } else {
219          //generate a token
220          final Token<? extends TokenIdentifier> t = generateDelegationToken(
221              namenode, ugi, request.getUserPrincipal().getName());
222          delegationQuery = "&" + new DelegationParam(t.encodeToUrlString());
223        }
224        final String query = op.toQueryString() + delegationQuery
225            + "&" + new NamenodeRpcAddressParam(namenode)
226            + Param.toSortedString("&", parameters);
227        final String uripath = WebHdfsFileSystem.PATH_PREFIX + path;
228    
229        final URI uri = new URI("http", null, dn.getHostName(), dn.getInfoPort(),
230            uripath, query, null);
231        if (LOG.isTraceEnabled()) {
232          LOG.trace("redirectURI=" + uri);
233        }
234        return uri;
235      }

350      private Response put(
351          final UserGroupInformation ugi,
352          final DelegationParam delegation,
353          final UserParam username,
354          final DoAsParam doAsUser,
355          final String fullpath,
356          final PutOpParam op,
357          final DestinationParam destination,
358          final OwnerParam owner,
359          final GroupParam group,
360          final PermissionParam permission,
361          final OverwriteParam overwrite,
362          final BufferSizeParam bufferSize,
363          final ReplicationParam replication,
364          final BlockSizeParam blockSize,
365          final ModificationTimeParam modificationTime,
366          final AccessTimeParam accessTime,
367          final RenameOptionSetParam renameOptions,
368          final CreateParentParam createParent,
369          final TokenArgumentParam delegationTokenArgument
370          ) throws IOException, URISyntaxException {
371    
372        final Configuration conf = (Configuration)context.getAttribute(JspHelper.CURRENT_CONF);
373        final NameNode namenode = (NameNode)context.getAttribute("name.node");
374        final NamenodeProtocols np = namenode.getRpcServer();
375    
376        switch(op.getValue()) {
377        case CREATE:
378        {
379          final URI uri = redirectURI(namenode, ugi, delegation, username, doAsUser,
380              fullpath, op.getValue(), -1L, blockSize.getValue(conf),
381              permission, overwrite, bufferSize, replication, blockSize);
382          return Response.temporaryRedirect(uri).type(MediaType.APPLICATION_OCTET_STREAM).build();
383        } 
384        case MKDIRS:
385        {
386          final boolean b = np.mkdirs(fullpath, permission.getFsPermission(), true);
387          final String js = JsonUtil.toJsonString("boolean", b);
388          return Response.ok(js).type(MediaType.APPLICATION_JSON).build();
389        }
390        case CREATESYMLINK:
391        {
392          np.createSymlink(destination.getValue(), fullpath,
393              PermissionParam.getDefaultFsPermission(), createParent.getValue());
394          return Response.ok().type(MediaType.APPLICATION_OCTET_STREAM).build();
395        }
396        case RENAME:
397        {
398          final EnumSet<Options.Rename> s = renameOptions.getValue();
399          if (s.isEmpty()) {
400            final boolean b = np.rename(fullpath, destination.getValue());
401            final String js = JsonUtil.toJsonString("boolean", b);
402            return Response.ok(js).type(MediaType.APPLICATION_JSON).build();
403          } else {
404            np.rename2(fullpath, destination.getValue(),
405                s.toArray(new Options.Rename[s.size()]));
406            return Response.ok().type(MediaType.APPLICATION_OCTET_STREAM).build();
407          }
408        }
409        case SETREPLICATION:
410        {
411          final boolean b = np.setReplication(fullpath, replication.getValue(conf));
412          final String js = JsonUtil.toJsonString("boolean", b);
413          return Response.ok(js).type(MediaType.APPLICATION_JSON).build();
414        }
415        case SETOWNER:
416        {
417          if (owner.getValue() == null && group.getValue() == null) {
418            throw new IllegalArgumentException("Both owner and group are empty.");
419          }
420    
421          np.setOwner(fullpath, owner.getValue(), group.getValue());
422          return Response.ok().type(MediaType.APPLICATION_OCTET_STREAM).build();
423        }
424        case SETPERMISSION:
425        {
426          np.setPermission(fullpath, permission.getFsPermission());
427          return Response.ok().type(MediaType.APPLICATION_OCTET_STREAM).build();
428        }
429        case SETTIMES:
430        {
431          np.setTimes(fullpath, modificationTime.getValue(), accessTime.getValue());
432          return Response.ok().type(MediaType.APPLICATION_OCTET_STREAM).build();
433        }
434        case RENEWDELEGATIONTOKEN:
435        {
436          final Token<DelegationTokenIdentifier> token = new Token<DelegationTokenIdentifier>();
437          token.decodeFromUrlString(delegationTokenArgument.getValue());
438          final long expiryTime = np.renewDelegationToken(token);
439          final String js = JsonUtil.toJsonString("long", expiryTime);
440          return Response.ok(js).type(MediaType.APPLICATION_JSON).build();
441        }
442        case CANCELDELEGATIONTOKEN:
443        {
444          final Token<DelegationTokenIdentifier> token = new Token<DelegationTokenIdentifier>();
445          token.decodeFromUrlString(delegationTokenArgument.getValue());
446          np.cancelDelegationToken(token);
447          return Response.ok().type(MediaType.APPLICATION_OCTET_STREAM).build();
448        }
449        default:
450          throw new UnsupportedOperationException(op + " is not supported");
451        }
452      }

566          @Context final UserGroupInformation ugi,

597              REMOTE_ADDRESS.set(null);
598            }
599          }

603      private Response get(
604          final UserGroupInformation ugi,
605          final DelegationParam delegation,
606          final UserParam username,
607          final DoAsParam doAsUser,
608          final String fullpath,
609          final GetOpParam op,
610          final OffsetParam offset,
611          final LengthParam length,
612          final RenewerParam renewer,
613          final BufferSizeParam bufferSize
614          ) throws IOException, URISyntaxException {
615        final NameNode namenode = (NameNode)context.getAttribute("name.node");
616        final NamenodeProtocols np = namenode.getRpcServer();
617    
618        switch(op.getValue()) {
619        case OPEN:
620        {
621          final URI uri = redirectURI(namenode, ugi, delegation, username, doAsUser,
622              fullpath, op.getValue(), offset.getValue(), -1L, offset, length, bufferSize);
623          return Response.temporaryRedirect(uri).type(MediaType.APPLICATION_OCTET_STREAM).build();
624        }
625        case GET_BLOCK_LOCATIONS:
626        {
627          final long offsetValue = offset.getValue();
628          final Long lengthValue = length.getValue();
629          final LocatedBlocks locatedblocks = np.getBlockLocations(fullpath,
630              offsetValue, lengthValue != null? lengthValue: Long.MAX_VALUE);
631          final String js = JsonUtil.toJsonString(locatedblocks);
632          return Response.ok(js).type(MediaType.APPLICATION_JSON).build();
633        }
634        case GETFILESTATUS:
635        {
636          final HdfsFileStatus status = np.getFileInfo(fullpath);
637          if (status == null) {
638            throw new FileNotFoundException("File does not exist: " + fullpath);
639          }
640    
641          final String js = JsonUtil.toJsonString(status, true);
642          return Response.ok(js).type(MediaType.APPLICATION_JSON).build();
643        }
644        case LISTSTATUS:
645        {
646          final StreamingOutput streaming = getListingStream(np, fullpath);
647          return Response.ok(streaming).type(MediaType.APPLICATION_JSON).build();
648        }
649        case GETCONTENTSUMMARY:
650        {
651          final ContentSummary contentsummary = np.getContentSummary(fullpath);
652          final String js = JsonUtil.toJsonString(contentsummary);
653          return Response.ok(js).type(MediaType.APPLICATION_JSON).build();
654        }
655        case GETFILECHECKSUM:
656        {
657          final URI uri = redirectURI(namenode, ugi, delegation, username, doAsUser,
658              fullpath, op.getValue(), -1L, -1L);
659          return Response.temporaryRedirect(uri).type(MediaType.APPLICATION_OCTET_STREAM).build();
660        }
661        case GETDELEGATIONTOKEN:
662        {
663          if (delegation.getValue() != null) {
664            throw new IllegalArgumentException(delegation.getName()
665                + " parameter is not null.");
666          }
667          final Token<? extends TokenIdentifier> token = generateDelegationToken(
668              namenode, ugi, renewer.getValue());
669          final String js = JsonUtil.toJsonString(token);
670          return Response.ok(js).type(MediaType.APPLICATION_JSON).build();
671        }
672        case GETHOMEDIRECTORY:
673        {
674          final String js = JsonUtil.toJsonString(
675              org.apache.hadoop.fs.Path.class.getSimpleName(),
676              WebHdfsFileSystem.getHomeDirectoryString(ugi));
677          return Response.ok(js).type(MediaType.APPLICATION_JSON).build();
678        }
679        default:
680          throw new UnsupportedOperationException(op + " is not supported");
681        }
682      }

684      private static DirectoryListing getDirectoryListing(final NamenodeProtocols np,
685          final String p, byte[] startAfter) throws IOException {
686        final DirectoryListing listing = np.getListing(p, startAfter, false);
687        if (listing == null) { // the directory does not exist
688          throw new FileNotFoundException("File " + p + " does not exist.");
689        }
690        return listing;
691      }

693      private static StreamingOutput getListingStream(final NamenodeProtocols np, 
694          final String p) throws IOException {
695        final DirectoryListing first = getDirectoryListing(np, p,
696            HdfsFileStatus.EMPTY_NAME);
697    
698        return new StreamingOutput() {
699          @Override
700          public void write(final OutputStream outstream) throws IOException {
701            final PrintStream out = new PrintStream(outstream);
702            out.println("{\"" + FileStatus.class.getSimpleName() + "es\":{\""
703                + FileStatus.class.getSimpleName() + "\":[");
704    
705            final HdfsFileStatus[] partial = first.getPartialListing();
706            if (partial.length > 0) {
707              out.print(JsonUtil.toJsonString(partial[0], false));
708            }
709            for(int i = 1; i < partial.length; i++) {
710              out.println(',');
711              out.print(JsonUtil.toJsonString(partial[i], false));
712            }
713    
714            for(DirectoryListing curr = first; curr.hasMore(); ) { 
715              curr = getDirectoryListing(np, p, curr.getLastName());
716              for(HdfsFileStatus s : curr.getPartialListing()) {
717                out.println(',');
718                out.print(JsonUtil.toJsonString(s, false));
719              }
720            }
721            
722            out.println();
723            out.println("]}}");
724          }
725        };
726      }
