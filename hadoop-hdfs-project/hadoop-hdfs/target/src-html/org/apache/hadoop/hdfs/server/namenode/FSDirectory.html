119      FSDirectory(FSImage fsImage, FSNamesystem ns, Configuration conf) {
120        this.dirLock = new ReentrantReadWriteLock(true); // fair
121        this.cond = dirLock.writeLock().newCondition();
122        rootDir = new INodeDirectoryWithQuota(INodeDirectory.ROOT_NAME,
123            ns.createFsOwnerPermissions(new FsPermission((short)0755)),
124            Integer.MAX_VALUE, UNKNOWN_DISK_SPACE);
125        this.fsImage = fsImage;
126        int configuredLimit = conf.getInt(
127            DFSConfigKeys.DFS_LIST_LIMIT, DFSConfigKeys.DFS_LIST_LIMIT_DEFAULT);
128        this.lsLimit = configuredLimit>0 ?
129            configuredLimit : DFSConfigKeys.DFS_LIST_LIMIT_DEFAULT;
130        
131        // filesystem limits
132        this.maxComponentLength = conf.getInt(
133            DFSConfigKeys.DFS_NAMENODE_MAX_COMPONENT_LENGTH_KEY,
134            DFSConfigKeys.DFS_NAMENODE_MAX_COMPONENT_LENGTH_DEFAULT);
135        this.maxDirItems = conf.getInt(
136            DFSConfigKeys.DFS_NAMENODE_MAX_DIRECTORY_ITEMS_KEY,
137            DFSConfigKeys.DFS_NAMENODE_MAX_DIRECTORY_ITEMS_DEFAULT);
138    
139        int threshold = conf.getInt(
140            DFSConfigKeys.DFS_NAMENODE_NAME_CACHE_THRESHOLD_KEY,
141            DFSConfigKeys.DFS_NAMENODE_NAME_CACHE_THRESHOLD_DEFAULT);
142        NameNode.LOG.info("Caching file names occuring more than " + threshold
143            + " times ");
144        nameCache = new NameCache<ByteArray>(threshold);
145        namesystem = ns;
146      }

195      /**
196       * Block until the object is ready to be used.
197       */
198      void waitForReady() {
199        if (!ready) {
200          writeLock();
201          try {
202            while (!ready) {
203              try {
204                cond.await(5000, TimeUnit.MILLISECONDS);
205              } catch (InterruptedException ie) {
206              }
207            }
208          } finally {
209            writeUnlock();
210          }
211        }
212      }

214      /**
215       * Add the given filename to the fs.
216       * @throws QuotaExceededException 
217       * @throws FileAlreadyExistsException 
218       */
219      INodeFileUnderConstruction addFile(String path, 
220                    PermissionStatus permissions,
221                    short replication,
222                    long preferredBlockSize,
223                    String clientName,
224                    String clientMachine,
225                    DatanodeDescriptor clientNode,
226                    long generationStamp) 
227        throws FileAlreadyExistsException, QuotaExceededException,
228          UnresolvedLinkException {
229        waitForReady();
230    
231        // Always do an implicit mkdirs for parent directory tree.
232        long modTime = now();
233        
234        Path parent = new Path(path).getParent();
235        if (parent == null) {
236          // Trying to add "/" as a file - this path has no
237          // parent -- avoids an NPE below.
238          return null;
239        }
240        
241        if (!mkdirs(parent.toString(), permissions, true, modTime)) {
242          return null;
243        }
244        INodeFileUnderConstruction newNode = new INodeFileUnderConstruction(
245                                     permissions,replication,
246                                     preferredBlockSize, modTime, clientName, 
247                                     clientMachine, clientNode);
248        writeLock();
249        try {
250          newNode = addNode(path, newNode, UNKNOWN_DISK_SPACE);
251        } finally {
252          writeUnlock();
253        }
254        if (newNode == null) {
255          NameNode.stateChangeLog.info("DIR* FSDirectory.addFile: "
256                                       +"failed to add "+path
257                                       +" to the file system");
258          return null;
259        }
260    
261        if(NameNode.stateChangeLog.isDebugEnabled()) {
262          NameNode.stateChangeLog.debug("DIR* FSDirectory.addFile: "
263              +path+" is added to the file system");
264        }
265        return newNode;
266      }

300      INodeDirectory addToParent(byte[] src, INodeDirectory parentINode,
301          INode newNode, boolean propagateModTime) throws UnresolvedLinkException {
302        // NOTE: This does not update space counts for parents
303        INodeDirectory newParent = null;
304        writeLock();
305        try {
306          try {
307            newParent = rootDir.addToParent(src, newNode, parentINode,
308                                            propagateModTime);
309            cacheName(newNode);
310          } catch (FileNotFoundException e) {
311            return null;
312          }
313          if(newParent == null)
314            return null;
315          if(!newNode.isDirectory() && !newNode.isLink()) {
316            // Add file->block mapping
317            INodeFile newF = (INodeFile)newNode;
318            BlockInfo[] blocks = newF.getBlocks();
319            for (int i = 0; i < blocks.length; i++) {
320              newF.setBlock(i, getBlockManager().addBlockCollection(blocks[i], newF));
321            }
322          }
323        } finally {
324          writeUnlock();
325        }
326        return newParent;
327      }

329      /**
330       * Add a block to the file. Returns a reference to the added block.
331       */
332      BlockInfo addBlock(String path,
333                         INode[] inodes,
334                         Block block,
335                         DatanodeDescriptor targets[]
336      ) throws QuotaExceededException {
337        waitForReady();
338    
339        writeLock();
340        try {
341          assert inodes[inodes.length-1].isUnderConstruction() :
342            "INode should correspond to a file under construction";
343          INodeFileUnderConstruction fileINode = 
344            (INodeFileUnderConstruction)inodes[inodes.length-1];
345    
346          // check quota limits and updated space consumed
347          updateCount(inodes, inodes.length-1, 0,
348              fileINode.getPreferredBlockSize()*fileINode.getReplication(), true);
349    
350          // associate new last block for the file
351          BlockInfoUnderConstruction blockInfo =
352            new BlockInfoUnderConstruction(
353                block,
354                fileINode.getReplication(),
355                BlockUCState.UNDER_CONSTRUCTION,
356                targets);
357          getBlockManager().addBlockCollection(blockInfo, fileINode);
358          fileINode.addBlock(blockInfo);
359    
360          if(NameNode.stateChangeLog.isDebugEnabled()) {
361            NameNode.stateChangeLog.debug("DIR* FSDirectory.addBlock: "
362                + path + " with " + block
363                + " block is added to the in-memory "
364                + "file system");
365          }
366          return blockInfo;
367        } finally {
368          writeUnlock();
369        }
370      }

497      /**
498       * Change a path name
499       * 
500       * @param src source path
501       * @param dst destination path
502       * @return true if rename succeeds; false otherwise
503       * @throws QuotaExceededException if the operation violates any quota limit
504       * @throws FileAlreadyExistsException if the src is a symlink that points to dst
505       * @deprecated See {@link #renameTo(String, String)}
506       */
507      @Deprecated
508      boolean unprotectedRenameTo(String src, String dst, long timestamp)
509        throws QuotaExceededException, UnresolvedLinkException, 
510        FileAlreadyExistsException {
511        assert hasWriteLock();
512        INode[] srcInodes = rootDir.getExistingPathINodes(src, false);
513        INode srcInode = srcInodes[srcInodes.length-1];
514        
515        // check the validation of the source
516        if (srcInode == null) {
517          NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
518              + "failed to rename " + src + " to " + dst
519              + " because source does not exist");
520          return false;
521        } 
522        if (srcInodes.length == 1) {
523          NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
524              +"failed to rename "+src+" to "+dst+ " because source is the root");
525          return false;
526        }
527        if (isDir(dst)) {
528          dst += Path.SEPARATOR + new Path(src).getName();
529        }
530        
531        // check the validity of the destination
532        if (dst.equals(src)) {
533          return true;
534        }
535        if (srcInode.isLink() && 
536            dst.equals(((INodeSymlink)srcInode).getLinkValue())) {
537          throw new FileAlreadyExistsException(
538              "Cannot rename symlink "+src+" to its target "+dst);
539        }
540        
541        // dst cannot be directory or a file under src
542        if (dst.startsWith(src) && 
543            dst.charAt(src.length()) == Path.SEPARATOR_CHAR) {
544          NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
545              + "failed to rename " + src + " to " + dst
546              + " because destination starts with src");
547          return false;
548        }
549        
550        byte[][] dstComponents = INode.getPathComponents(dst);
551        INode[] dstInodes = new INode[dstComponents.length];
552        rootDir.getExistingPathINodes(dstComponents, dstInodes, false);
553        if (dstInodes[dstInodes.length-1] != null) {
554          NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
555                                       +"failed to rename "+src+" to "+dst+ 
556                                       " because destination exists");
557          return false;
558        }
559        if (dstInodes[dstInodes.length-2] == null) {
560          NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
561              +"failed to rename "+src+" to "+dst+ 
562              " because destination's parent does not exist");
563          return false;
564        }
565        
566        // Ensure dst has quota to accommodate rename
567        verifyQuotaForRename(srcInodes,dstInodes);
568        
569        INode dstChild = null;
570        INode srcChild = null;
571        String srcChildName = null;
572        try {
573          // remove src
574          srcChild = removeChild(srcInodes, srcInodes.length-1);
575          if (srcChild == null) {
576            NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
577                + "failed to rename " + src + " to " + dst
578                + " because the source can not be removed");
579            return false;
580          }
581          srcChildName = srcChild.getLocalName();
582          srcChild.setLocalName(dstComponents[dstInodes.length-1]);
583          
584          // add src to the destination
585          dstChild = addChildNoQuotaCheck(dstInodes, dstInodes.length - 1,
586              srcChild, UNKNOWN_DISK_SPACE);
587          if (dstChild != null) {
588            srcChild = null;
589            if (NameNode.stateChangeLog.isDebugEnabled()) {
590              NameNode.stateChangeLog.debug("DIR* FSDirectory.unprotectedRenameTo: " 
591                  + src + " is renamed to " + dst);
592            }
593            // update modification time of dst and the parent of src
594            srcInodes[srcInodes.length-2].setModificationTime(timestamp);
595            dstInodes[dstInodes.length-2].setModificationTime(timestamp);
596            return true;
597          }
598        } finally {
599          if (dstChild == null && srcChild != null) {
600            // put it back
601            srcChild.setLocalName(srcChildName);
602            addChildNoQuotaCheck(srcInodes, srcInodes.length - 1, srcChild, 
603                UNKNOWN_DISK_SPACE);
604          }
605        }
606        NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
607            +"failed to rename "+src+" to "+dst);
608        return false;
609      }

611      /**
612       * Rename src to dst.
613       * See {@link DistributedFileSystem#rename(Path, Path, Options.Rename...)}
614       * for details related to rename semantics and exceptions.
615       * 
616       * @param src source path
617       * @param dst destination path
618       * @param timestamp modification time
619       * @param options Rename options
620       */
621      boolean unprotectedRenameTo(String src, String dst, long timestamp,
622          Options.Rename... options) throws FileAlreadyExistsException,
623          FileNotFoundException, ParentNotDirectoryException,
624          QuotaExceededException, UnresolvedLinkException, IOException {
625        assert hasWriteLock();
626        boolean overwrite = false;
627        if (null != options) {
628          for (Rename option : options) {
629            if (option == Rename.OVERWRITE) {
630              overwrite = true;
631            }
632          }
633        }
634        String error = null;
635        final INode[] srcInodes = rootDir.getExistingPathINodes(src, false);
636        final INode srcInode = srcInodes[srcInodes.length - 1];
637        // validate source
638        if (srcInode == null) {
639          error = "rename source " + src + " is not found.";
640          NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
641              + error);
642          throw new FileNotFoundException(error);
643        }
644        if (srcInodes.length == 1) {
645          error = "rename source cannot be the root";
646          NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
647              + error);
648          throw new IOException(error);
649        }
650    
651        // validate the destination
652        if (dst.equals(src)) {
653          throw new FileAlreadyExistsException(
654              "The source "+src+" and destination "+dst+" are the same");
655        }
656        if (srcInode.isLink() && 
657            dst.equals(((INodeSymlink)srcInode).getLinkValue())) {
658          throw new FileAlreadyExistsException(
659              "Cannot rename symlink "+src+" to its target "+dst);
660        }
661        // dst cannot be a directory or a file under src
662        if (dst.startsWith(src) && 
663            dst.charAt(src.length()) == Path.SEPARATOR_CHAR) {
664          error = "Rename destination " + dst
665              + " is a directory or file under source " + src;
666          NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
667              + error);
668          throw new IOException(error);
669        }
670        final byte[][] dstComponents = INode.getPathComponents(dst);
671        final INode[] dstInodes = new INode[dstComponents.length];
672        rootDir.getExistingPathINodes(dstComponents, dstInodes, false);
673        INode dstInode = dstInodes[dstInodes.length - 1];
674        if (dstInodes.length == 1) {
675          error = "rename destination cannot be the root";
676          NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
677              + error);
678          throw new IOException(error);
679        }
680        if (dstInode != null) { // Destination exists
681          // It's OK to rename a file to a symlink and vice versa
682          if (dstInode.isDirectory() != srcInode.isDirectory()) {
683            error = "Source " + src + " and destination " + dst
684                + " must both be directories";
685            NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
686                + error);
687            throw new IOException(error);
688          }
689          if (!overwrite) { // If destination exists, overwrite flag must be true
690            error = "rename destination " + dst + " already exists";
691            NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
692                + error);
693            throw new FileAlreadyExistsException(error);
694          }
695          List<INode> children = dstInode.isDirectory() ? 
696              ((INodeDirectory) dstInode).getChildrenRaw() : null;
697          if (children != null && children.size() != 0) {
698            error = "rename cannot overwrite non empty destination directory "
699                + dst;
700            NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
701                + error);
702            throw new IOException(error);
703          }
704        }
705        if (dstInodes[dstInodes.length - 2] == null) {
706          error = "rename destination parent " + dst + " not found.";
707          NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
708              + error);
709          throw new FileNotFoundException(error);
710        }
711        if (!dstInodes[dstInodes.length - 2].isDirectory()) {
712          error = "rename destination parent " + dst + " is a file.";
713          NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
714              + error);
715          throw new ParentNotDirectoryException(error);
716        }
717    
718        // Ensure dst has quota to accommodate rename
719        verifyQuotaForRename(srcInodes, dstInodes);
720        INode removedSrc = removeChild(srcInodes, srcInodes.length - 1);
721        if (removedSrc == null) {
722          error = "Failed to rename " + src + " to " + dst
723              + " because the source can not be removed";
724          NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
725              + error);
726          throw new IOException(error);
727        }
728        final String srcChildName = removedSrc.getLocalName();
729        String dstChildName = null;
730        INode removedDst = null;
731        try {
732          if (dstInode != null) { // dst exists remove it
733            removedDst = removeChild(dstInodes, dstInodes.length - 1);
734            dstChildName = removedDst.getLocalName();
735          }
736    
737          INode dstChild = null;
738          removedSrc.setLocalName(dstComponents[dstInodes.length - 1]);
739          // add src as dst to complete rename
740          dstChild = addChildNoQuotaCheck(dstInodes, dstInodes.length - 1,
741              removedSrc, UNKNOWN_DISK_SPACE);
742    
743          int filesDeleted = 0;
744          if (dstChild != null) {
745            removedSrc = null;
746            if (NameNode.stateChangeLog.isDebugEnabled()) {
747              NameNode.stateChangeLog.debug(
748                  "DIR* FSDirectory.unprotectedRenameTo: " + src
749                  + " is renamed to " + dst);
750            }
751            srcInodes[srcInodes.length - 2].setModificationTime(timestamp);
752            dstInodes[dstInodes.length - 2].setModificationTime(timestamp);
753    
754            // Collect the blocks and remove the lease for previous dst
755            if (removedDst != null) {
756              INode rmdst = removedDst;
757              removedDst = null;
758              List<Block> collectedBlocks = new ArrayList<Block>();
759              filesDeleted = rmdst.collectSubtreeBlocksAndClear(collectedBlocks);
760              getFSNamesystem().removePathAndBlocks(src, collectedBlocks);
761            }
762            return filesDeleted >0;
763          }
764        } finally {
765          if (removedSrc != null) {
766            // Rename failed - restore src
767            removedSrc.setLocalName(srcChildName);
768            addChildNoQuotaCheck(srcInodes, srcInodes.length - 1, removedSrc, 
769                UNKNOWN_DISK_SPACE);
770          }
771          if (removedDst != null) {
772            // Rename failed - restore dst
773            removedDst.setLocalName(dstChildName);
774            addChildNoQuotaCheck(dstInodes, dstInodes.length - 1, removedDst, 
775                UNKNOWN_DISK_SPACE);
776          }
777        }
778        NameNode.stateChangeLog.warn("DIR* FSDirectory.unprotectedRenameTo: "
779            + "failed to rename " + src + " to " + dst);
780        throw new IOException("rename from " + src + " to " + dst + " failed.");
781      }

947       * Must be public because also called from EditLogs
948       * NOTE: - it does not update quota (not needed for concat)
949       */

951          throws UnresolvedLinkException {
952        assert hasWriteLock();
953        if (NameNode.stateChangeLog.isDebugEnabled()) {
954          NameNode.stateChangeLog.debug("DIR* FSNamesystem.concat to "+target);

957

959        INodeFile trgInode = (INodeFile) trgINodes[trgINodes.length-1];
960        INodeDirectory trgParent = (INodeDirectory)trgINodes[trgINodes.length-2];
961

966          INodeFile srcInode = getFileINode(src);
967          allSrcInodes[i++] = srcInode;
968          totalBlocks += srcInode.blocks.length;  
969        }
970        trgInode.appendBlocks(allSrcInodes, totalBlocks); // copy the blocks
971        
972        // since we are in the same dir - we can use same parent to remove files
973        int count = 0;
974        for(INodeFile nodeToRemove: allSrcInodes) {
975          if(nodeToRemove == null) continue;
976          
977          nodeToRemove.blocks = null;
978          trgParent.removeChild(nodeToRemove);
979          count++;
980        }
981        
982        trgInode.setModificationTimeForce(timestamp);
983        trgParent.setModificationTime(timestamp);
984        // update quota on the parent directory ('count' files removed, 0 space)
985        unprotectedUpdateCount(trgINodes, trgINodes.length-1, - count, 0);

1363                               throws QuotaExceededException {

1376          if (inodes[i].isQuotaSet()) { // a directory with quota
1377            INodeDirectoryWithQuota node =(INodeDirectoryWithQuota)inodes[i]; 
1378            node.updateNumItemsInTree(nsDelta, dsDelta);
1379          }
1380        }
1381      }

1431      /** Return the full path name of the specified inode */
1432      static String getFullPathName(INode inode) {
1433        // calculate the depth of this inode from root
1434        int depth = 0;
1435        for (INode i = inode; i != null; i = i.parent) {
1436          depth++;
1437        }
1438        INode[] inodes = new INode[depth];
1439    
1440        // fill up the inodes in the path from this inode to root
1441        for (int i = 0; i < depth; i++) {
1442          inodes[depth-i-1] = inode;
1443          inode = inode.parent;
1444        }
1445        return getFullPathName(inodes, depth-1);

1448      /**
1449       * Create a directory 
1450       * If ancestor directories do not exist, automatically create them.
1451    
1452       * @param src string representation of the path to the directory
1453       * @param permissions the permission of the directory
1454       * @param isAutocreate if the permission of the directory should inherit
1455       *                          from its parent or not. u+wx is implicitly added to
1456       *                          the automatically created directories, and to the
1457       *                          given directory if inheritPermission is true
1458       * @param now creation time
1459       * @return true if the operation succeeds false otherwise
1460       * @throws FileNotFoundException if an ancestor or itself is a file
1461       * @throws QuotaExceededException if directory creation violates 
1462       *                                any quota limit
1463       * @throws UnresolvedLinkException if a symlink is encountered in src.                      
1464       */
1465      boolean mkdirs(String src, PermissionStatus permissions,
1466          boolean inheritPermission, long now)
1467          throws FileAlreadyExistsException, QuotaExceededException, 
1468                 UnresolvedLinkException {
1469        src = normalizePath(src);
1470        String[] names = INode.getPathNames(src);
1471        byte[][] components = INode.getPathComponents(names);
1472        INode[] inodes = new INode[components.length];
1473        final int lastInodeIndex = inodes.length - 1;
1474    
1475        writeLock();
1476        try {
1477          rootDir.getExistingPathINodes(components, inodes, false);
1478    
1479          // find the index of the first null in inodes[]
1480          StringBuilder pathbuilder = new StringBuilder();
1481          int i = 1;
1482          for(; i < inodes.length && inodes[i] != null; i++) {
1483            pathbuilder.append(Path.SEPARATOR + names[i]);
1484            if (!inodes[i].isDirectory()) {
1485              throw new FileAlreadyExistsException("Parent path is not a directory: "
1486                  + pathbuilder+ " "+inodes[i].getLocalName());
1487            }
1488          }
1489    
1490          // default to creating parent dirs with the given perms
1491          PermissionStatus parentPermissions = permissions;
1492    
1493          // if not inheriting and it's the last inode, there's no use in
1494          // computing perms that won't be used
1495          if (inheritPermission || (i < lastInodeIndex)) {
1496            // if inheriting (ie. creating a file or symlink), use the parent dir,
1497            // else the supplied permissions
1498            // NOTE: the permissions of the auto-created directories violate posix
1499            FsPermission parentFsPerm = inheritPermission
1500                ? inodes[i-1].getFsPermission() : permissions.getPermission();
1501            
1502            // ensure that the permissions allow user write+execute
1503            if (!parentFsPerm.getUserAction().implies(FsAction.WRITE_EXECUTE)) {
1504              parentFsPerm = new FsPermission(
1505                  parentFsPerm.getUserAction().or(FsAction.WRITE_EXECUTE),
1506                  parentFsPerm.getGroupAction(),
1507                  parentFsPerm.getOtherAction()
1508              );
1509            }
1510            
1511            if (!parentPermissions.getPermission().equals(parentFsPerm)) {
1512              parentPermissions = new PermissionStatus(
1513                  parentPermissions.getUserName(),
1514                  parentPermissions.getGroupName(),
1515                  parentFsPerm
1516              );
1517              // when inheriting, use same perms for entire path
1518              if (inheritPermission) permissions = parentPermissions;
1519            }
1520          }
1521          
1522          // create directories beginning from the first null index
1523          for(; i < inodes.length; i++) {
1524            pathbuilder.append(Path.SEPARATOR + names[i]);
1525            String cur = pathbuilder.toString();
1526            unprotectedMkdir(inodes, i, components[i],
1527                (i < lastInodeIndex) ? parentPermissions : permissions, now);
1528            if (inodes[i] == null) {
1529              return false;
1530            }
1531            // Directory creation also count towards FilesCreated
1532            // to match count of FilesDeleted metric.
1533            if (getFSNamesystem() != null)
1534              NameNode.getNameNodeMetrics().incrFilesCreated();
1535            fsImage.getEditLog().logMkDir(cur, inodes[i]);
1536            if(NameNode.stateChangeLog.isDebugEnabled()) {
1537              NameNode.stateChangeLog.debug(
1538                  "DIR* FSDirectory.mkdirs: created directory " + cur);
1539            }
1540          }
1541        } finally {
1542          writeUnlock();
1543        }
1544        return true;
1545      }

1595      /**
1596       * Verify quota for adding or moving a new INode with required 
1597       * namespace and diskspace to a given position.
1598       *  
1599       * @param inodes INodes corresponding to a path
1600       * @param pos position where a new INode will be added
1601       * @param nsDelta needed namespace
1602       * @param dsDelta needed diskspace
1603       * @param commonAncestor Last node in inodes array that is a common ancestor
1604       *          for a INode that is being moved from one location to the other.
1605       *          Pass null if a node is not being moved.
1606       * @throws QuotaExceededException if quota limit is exceeded.
1607       */
1608      private void verifyQuota(INode[] inodes, int pos, long nsDelta, long dsDelta,
1609          INode commonAncestor) throws QuotaExceededException {
1610        if (!ready) {
1611          // Do not check quota if edits log is still being processed
1612          return;
1613        }
1614        if (nsDelta <= 0 && dsDelta <= 0) {
1615          // if quota is being freed or not being consumed
1616          return;
1617        }
1618        if (pos>inodes.length) {
1619          pos = inodes.length;
1620        }
1621        int i = pos - 1;
1622        try {
1623          // check existing components in the path  
1624          for(; i >= 0; i--) {
1625            if (commonAncestor == inodes[i]) {
1626              // Moving an existing node. Stop checking for quota when common
1627              // ancestor is reached
1628              return;
1629            }
1630            if (inodes[i].isQuotaSet()) { // a directory with quota
1631              INodeDirectoryWithQuota node =(INodeDirectoryWithQuota)inodes[i]; 
1632              node.verifyQuota(nsDelta, dsDelta);
1633            }
1634          }
1635        } catch (QuotaExceededException e) {
1636          e.setPathName(getFullPathName(inodes, i));
1637          throw e;
1638        }
1639      }

1641      /**
1642       * Verify quota for rename operation where srcInodes[srcInodes.length-1] moves
1643       * dstInodes[dstInodes.length-1]
1644       * 
1645       * @param srcInodes directory from where node is being moved.
1646       * @param dstInodes directory to where node is moved to.
1647       * @throws QuotaExceededException if quota limit is exceeded.
1648       */
1649      private void verifyQuotaForRename(INode[] srcInodes, INode[]dstInodes)
1650          throws QuotaExceededException {
1651        if (!ready) {
1652          // Do not check quota if edits log is still being processed
1653          return;
1654        }
1655        INode srcInode = srcInodes[srcInodes.length - 1];
1656        INode commonAncestor = null;
1657        for(int i =0;srcInodes[i] == dstInodes[i]; i++) {
1658          commonAncestor = srcInodes[i];
1659        }
1660        INode.DirCounts srcCounts = new INode.DirCounts();
1661        srcInode.spaceConsumedInTree(srcCounts);
1662        long nsDelta = srcCounts.getNsCount();
1663        long dsDelta = srcCounts.getDsCount();
1664        
1665        // Reduce the required quota by dst that is being removed
1666        INode dstInode = dstInodes[dstInodes.length - 1];
1667        if (dstInode != null) {
1668          INode.DirCounts dstCounts = new INode.DirCounts();
1669          dstInode.spaceConsumedInTree(dstCounts);
1670          nsDelta -= dstCounts.getNsCount();
1671          dsDelta -= dstCounts.getDsCount();
1672        }
1673        verifyQuota(dstInodes, dstInodes.length - 1, nsDelta, dsDelta,
1674            commonAncestor);
1675      }

1677      /**
1678       * Verify that filesystem limit constraints are not violated
1679       * @throws PathComponentTooLongException child's name is too long
1680       * @throws MaxDirectoryItemsExceededException items per directory is exceeded
1681       */
1682      protected <T extends INode> void verifyFsLimits(INode[] pathComponents,
1683          int pos, T child) throws FSLimitException {
1684        boolean includeChildName = false;
1685        try {
1686          if (maxComponentLength != 0) {
1687            int length = child.getLocalName().length();
1688            if (length > maxComponentLength) {
1689              includeChildName = true;
1690              throw new PathComponentTooLongException(maxComponentLength, length);
1691            }
1692          }
1693          if (maxDirItems != 0) {
1694            INodeDirectory parent = (INodeDirectory)pathComponents[pos-1];
1695            int count = parent.getChildren().size();
1696            if (count >= maxDirItems) {
1697              throw new MaxDirectoryItemsExceededException(maxDirItems, count);
1698            }
1699          }
1700        } catch (FSLimitException e) {
1701          String badPath = getFullPathName(pathComponents, pos-1);
1702          if (includeChildName) {
1703            badPath += Path.SEPARATOR + child.getLocalName();
1704          }
1705          e.setPathName(badPath);
1706          // Do not throw if edits log is still being processed
1707          if (ready) throw(e);
1708          // log pre-existing paths that exceed limits
1709          NameNode.LOG.error("FSDirectory.verifyFsLimits - " + e.getLocalizedMessage());
1710        }
1711      }

1713      /** Add a node child to the inodes at index pos. 
1714       * Its ancestors are stored at [0, pos-1]. 
1715       * QuotaExceededException is thrown if it violates quota limit */
1716      private <T extends INode> T addChild(INode[] pathComponents, int pos,
1717          T child, long childDiskspace,
1718          boolean checkQuota) throws QuotaExceededException {
1719            // The filesystem limits are not really quotas, so this check may appear
1720            // odd.  It's because a rename operation deletes the src, tries to add
1721            // to the dest, if that fails, re-adds the src from whence it came.
1722            // The rename code disables the quota when it's restoring to the
1723            // original location becase a quota violation would cause the the item
1724            // to go "poof".  The fs limits must be bypassed for the same reason.
1725        if (checkQuota) {
1726          verifyFsLimits(pathComponents, pos, child);
1727        }
1728        
1729        INode.DirCounts counts = new INode.DirCounts();
1730        child.spaceConsumedInTree(counts);
1731        if (childDiskspace < 0) {
1732          childDiskspace = counts.getDsCount();
1733        }
1734        updateCount(pathComponents, pos, counts.getNsCount(), childDiskspace,
1735            checkQuota);
1736        if (pathComponents[pos-1] == null) {
1737          throw new NullPointerException("Panic: parent does not exist");
1738        }
1739        T addedNode = ((INodeDirectory)pathComponents[pos-1]).addChild(
1740            child, true);
1741        if (addedNode == null) {
1742          updateCount(pathComponents, pos, -counts.getNsCount(), 
1743              -childDiskspace, true);
1744        }
1745        return addedNode;
1746      }

1820      /** 
1821       * Update the count of the directory if it has a quota and return the count
1822       * 
1823       * This does not throw a QuotaExceededException. This is just an update
1824       * of of existing state and throwing QuotaExceededException does not help
1825       * with fixing the state, if there is a problem.
1826       * 
1827       * @param dir the root of the tree that represents the directory
1828       * @param counters counters for name space and disk space
1829       * @param nodesInPath INodes for the each of components in the path.
1830       */
1831      private static void updateCountForINodeWithQuota(INodeDirectory dir, 
1832                                                   INode.DirCounts counts,
1833                                                   ArrayList<INode> nodesInPath) {
1834        long parentNamespace = counts.nsCount;
1835        long parentDiskspace = counts.dsCount;
1836        
1837        counts.nsCount = 1L;//for self. should not call node.spaceConsumedInTree()
1838        counts.dsCount = 0L;
1839        
1840        /* We don't need nodesInPath if we could use 'parent' field in 
1841         * INode. using 'parent' is not currently recommended. */
1842        nodesInPath.add(dir);
1843    
1844        for (INode child : dir.getChildren()) {
1845          if (child.isDirectory()) {
1846            updateCountForINodeWithQuota((INodeDirectory)child, 
1847                                         counts, nodesInPath);
1848          } else if (child.isLink()) {
1849            counts.nsCount += 1;
1850          } else { // reduce recursive calls
1851            counts.nsCount += 1;
1852            counts.dsCount += ((INodeFile)child).diskspaceConsumed();
1853          }
1854        }
1855          
1856        if (dir.isQuotaSet()) {
1857          ((INodeDirectoryWithQuota)dir).setSpaceConsumed(counts.nsCount,
1858                                                          counts.dsCount);
1859    
1860          // check if quota is violated for some reason.
1861          if ((dir.getNsQuota() >= 0 && counts.nsCount > dir.getNsQuota()) ||
1862              (dir.getDsQuota() >= 0 && counts.dsCount > dir.getDsQuota())) {
1863    
1864            // can only happen because of a software bug. the bug should be fixed.
1865            StringBuilder path = new StringBuilder(512);
1866            for (INode n : nodesInPath) {
1867              path.append('/');
1868              path.append(n.getLocalName());
1869            }
1870            
1871            NameNode.LOG.warn("Quota violation in image for " + path + 
1872                              " (Namespace quota : " + dir.getNsQuota() +
1873                              " consumed : " + counts.nsCount + ")" +
1874                              " (Diskspace quota : " + dir.getDsQuota() +
1875                              " consumed : " + counts.dsCount + ").");
1876          }            
1877        }
1878          
1879        // pop 
1880        nodesInPath.remove(nodesInPath.size()-1);
1881        
1882        counts.nsCount += parentNamespace;
1883        counts.dsCount += parentDiskspace;
1884      }

1886      /**
1887       * See {@link ClientProtocol#setQuota(String, long, long)} for the contract.
1888       * Sets quota for for a directory.
1889       * @returns INodeDirectory if any of the quotas have changed. null other wise.
1890       * @throws FileNotFoundException if the path does not exist or is a file
1891       * @throws QuotaExceededException if the directory tree size is 
1892       *                                greater than the given quota
1893       * @throws UnresolvedLinkException if a symlink is encountered in src.
1894       */
1895      INodeDirectory unprotectedSetQuota(String src, long nsQuota, long dsQuota)
1896        throws FileNotFoundException, QuotaExceededException, 
1897          UnresolvedLinkException {
1898        assert hasWriteLock();
1899        // sanity check
1900        if ((nsQuota < 0 && nsQuota != HdfsConstants.QUOTA_DONT_SET && 
1901             nsQuota < HdfsConstants.QUOTA_RESET) || 
1902            (dsQuota < 0 && dsQuota != HdfsConstants.QUOTA_DONT_SET && 
1903              dsQuota < HdfsConstants.QUOTA_RESET)) {
1904          throw new IllegalArgumentException("Illegal value for nsQuota or " +
1905                                             "dsQuota : " + nsQuota + " and " +
1906                                             dsQuota);
1907        }
1908        
1909        String srcs = normalizePath(src);
1910    
1911        INode[] inodes = rootDir.getExistingPathINodes(src, true);
1912        INode targetNode = inodes[inodes.length-1];
1913        if (targetNode == null) {
1914          throw new FileNotFoundException("Directory does not exist: " + srcs);
1915        } else if (!targetNode.isDirectory()) {
1916          throw new FileNotFoundException("Cannot set quota on a file: " + srcs);  
1917        } else if (targetNode.isRoot() && nsQuota == HdfsConstants.QUOTA_RESET) {
1918          throw new IllegalArgumentException("Cannot clear namespace quota on root.");
1919        } else { // a directory inode
1920          INodeDirectory dirNode = (INodeDirectory)targetNode;
1921          long oldNsQuota = dirNode.getNsQuota();
1922          long oldDsQuota = dirNode.getDsQuota();
1923          if (nsQuota == HdfsConstants.QUOTA_DONT_SET) {
1924            nsQuota = oldNsQuota;
1925          }
1926          if (dsQuota == HdfsConstants.QUOTA_DONT_SET) {
1927            dsQuota = oldDsQuota;
1928          }        
1929    
1930          if (dirNode instanceof INodeDirectoryWithQuota) { 
1931            // a directory with quota; so set the quota to the new value
1932            ((INodeDirectoryWithQuota)dirNode).setQuota(nsQuota, dsQuota);
1933            if (!dirNode.isQuotaSet()) {
1934              // will not come here for root because root's nsQuota is always set
1935              INodeDirectory newNode = new INodeDirectory(dirNode);
1936              INodeDirectory parent = (INodeDirectory)inodes[inodes.length-2];
1937              dirNode = newNode;
1938              parent.replaceChild(newNode);
1939            }
1940          } else {
1941            // a non-quota directory; so replace it with a directory with quota
1942            INodeDirectoryWithQuota newNode = 
1943              new INodeDirectoryWithQuota(nsQuota, dsQuota, dirNode);
1944            // non-root directory node; parent != null
1945            INodeDirectory parent = (INodeDirectory)inodes[inodes.length-2];
1946            dirNode = newNode;
1947            parent.replaceChild(newNode);
1948          }
1949          return (oldNsQuota != nsQuota || oldDsQuota != dsQuota) ? dirNode : null;
1950        }
1951      }

2005      private boolean unprotectedSetTimes(String src, INode inode, long mtime,
2006                                          long atime, boolean force) {
2007        assert hasWriteLock();
2008        boolean status = false;
2009        if (mtime != -1) {
2010          inode.setModificationTimeForce(mtime);
2011          status = true;
2012        }
2013        if (atime != -1) {
2014          long inodeTime = inode.getAccessTime();
2015    
2016          // if the last access time update was within the last precision interval, then
2017          // no need to store access time
2018          if (atime <= inodeTime + getFSNamesystem().getAccessTimePrecision() && !force) {
2019            status =  false;
2020          } else {
2021            inode.setAccessTime(atime);
2022            status = true;
2023          }
2024        } 
2025        return status;
2026      }

2119      /**
2120       * Add the given symbolic link to the fs. Record it in the edits log.
2121       */
2122      INodeSymlink addSymlink(String path, String target,
2123          PermissionStatus dirPerms, boolean createParent)
2124          throws UnresolvedLinkException, FileAlreadyExistsException,
2125          QuotaExceededException, IOException {
2126        waitForReady();
2127    
2128        final long modTime = now();
2129        if (createParent) {
2130          final String parent = new Path(path).getParent().toString();
2131          if (!mkdirs(parent, dirPerms, true, modTime)) {
2132            return null;
2133          }
2134        }
2135        final String userName = dirPerms.getUserName();
2136        INodeSymlink newNode  = null;
2137        writeLock();
2138        try {
2139          newNode = unprotectedSymlink(path, target, modTime, modTime,
2140              new PermissionStatus(userName, null, FsPermission.getDefault()));
2141        } finally {
2142          writeUnlock();
2143        }
2144        if (newNode == null) {
2145          NameNode.stateChangeLog.info("DIR* FSDirectory.addSymlink: "
2146                                       +"failed to add "+path
2147                                       +" to the file system");
2148          return null;
2149        }
2150        fsImage.getEditLog().logSymlink(path, target, modTime, modTime, newNode);
2151        
2152        if(NameNode.stateChangeLog.isDebugEnabled()) {
2153          NameNode.stateChangeLog.debug("DIR* FSDirectory.addSymlink: "
2154              +path+" is added to the file system");
2155        }
2156        return newNode;
2157      }

2159      /**
2160       * Add the specified path into the namespace. Invoked from edit log processing.
2161       */
2162      INodeSymlink unprotectedSymlink(String path, String target, long modTime, 
2163                                      long atime, PermissionStatus perm) 
2164          throws UnresolvedLinkException {
2165        assert hasWriteLock();
2166        INodeSymlink newNode = new INodeSymlink(target, modTime, atime, perm);
2167        try {
2168          newNode = addNode(path, newNode, UNKNOWN_DISK_SPACE);
2169        } catch (UnresolvedLinkException e) {
2170          /* All UnresolvedLinkExceptions should have been resolved by now, but we
2171           * should re-throw them in case that changes so they are not swallowed 
2172           * by catching IOException below.
2173           */
2174          throw e;
2175        } catch (IOException e) {
2176          return null;
2177        }
2178        return newNode;
2179      }
