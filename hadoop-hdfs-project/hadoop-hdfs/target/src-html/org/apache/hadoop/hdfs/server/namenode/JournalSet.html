049     * assumed that FSEditLog methods, that use this class, use proper

056      static final public Comparator<EditLogInputStream>
057        EDIT_LOG_INPUT_STREAM_COMPARATOR = new Comparator<EditLogInputStream>() {
058          @Override
059          public int compare(EditLogInputStream a, EditLogInputStream b) {
060            return ComparisonChain.start().
061              compare(a.getFirstTxId(), b.getFirstTxId()).
062              compare(b.getLastTxId(), a.getLastTxId()).
063              result();
064          }
065        };

067      /**
068       * Container for a JournalManager paired with its currently
069       * active stream.
070       * 
071       * If a Journal gets disabled due to an error writing to its
072       * stream, then the stream will be aborted and set to null.
073       */
074      static class JournalAndStream implements CheckableNameNodeResource {
075        private final JournalManager journal;
076        private boolean disabled = false;
077        private EditLogOutputStream stream;
078        private boolean required = false;
079        
080        public JournalAndStream(JournalManager manager, boolean required) {
081          this.journal = manager;
082          this.required = required;
083        }
084    
085        public void startLogSegment(long txId) throws IOException {
086          Preconditions.checkState(stream == null);
087          disabled = false;
088          stream = journal.startLogSegment(txId);
089        }
090    
091        /**
092         * Closes the stream, also sets it to null.
093         */
094        public void closeStream() throws IOException {
095          if (stream == null) return;
096          stream.close();
097          stream = null;
098        }
099    
100        /**
101         * Close the Journal and Stream
102         */
103        public void close() throws IOException {
104          closeStream();
105    
106          journal.close();
107        }
108        
109        /**
110         * Aborts the stream, also sets it to null.
111         */
112        public void abort() {
113          if (stream == null) return;
114          try {
115            stream.abort();
116          } catch (IOException ioe) {
117            LOG.error("Unable to abort stream " + stream, ioe);
118          }
119          stream = null;
120        }
121    
122        boolean isActive() {
123          return stream != null;
124        }
125        
126        /**
127         * Should be used outside JournalSet only for testing.
128         */
129        EditLogOutputStream getCurrentStream() {
130          return stream;
131        }
132        
133        @Override
134        public String toString() {
135          return "JournalAndStream(mgr=" + journal +
136            ", " + "stream=" + stream + ")";
137        }

139        void setCurrentStreamForTests(EditLogOutputStream stream) {
140          this.stream = stream;
141        }

144          return journal;
145        }

167      final int minimumRedundantJournals;

173      @Override
174      public EditLogOutputStream startLogSegment(final long txId) throws IOException {

176          @Override
177          public void apply(JournalAndStream jas) throws IOException {

189          public void apply(JournalAndStream jas) throws IOException {
190            if (jas.isActive()) {
191              jas.closeStream();
192              jas.getManager().finalizeLogSegment(firstTxId, lastTxId);

194          }
195        }, "finalize log segment " + firstTxId + ", " + lastTxId);

204          }

208      /**
209       * In this function, we get a bunch of streams from all of our JournalManager
210       * objects.  Then we add these to the collection one by one.
211       * 
212       * @param streams          The collection to add the streams to.  It may or 
213       *                         may not be sorted-- this is up to the caller.
214       * @param fromTxId         The transaction ID to start looking for streams at
215       * @param inProgressOk     Should we consider unfinalized streams?
216       */
217      @Override
218      public void selectInputStreams(Collection<EditLogInputStream> streams,
219          long fromTxId, boolean inProgressOk) {
220        final TreeMultiset<EditLogInputStream> allStreams =
221            TreeMultiset.create(EDIT_LOG_INPUT_STREAM_COMPARATOR);
222        for (JournalAndStream jas : journals) {
223          if (jas.isDisabled()) {
224            LOG.info("Skipping jas " + jas + " since it's disabled");
225            continue;
226          }
227          jas.getManager().selectInputStreams(allStreams, fromTxId, inProgressOk);
228        }
229        // We want to group together all the streams that start on the same start
230        // transaction ID.  To do this, we maintain an accumulator (acc) of all
231        // the streams we've seen at a given start transaction ID.  When we see a
232        // higher start transaction ID, we select a stream from the accumulator and
233        // clear it.  Then we begin accumulating streams with the new, higher start
234        // transaction ID.
235        LinkedList<EditLogInputStream> acc =
236            new LinkedList<EditLogInputStream>();
237        for (EditLogInputStream elis : allStreams) {
238          if (acc.isEmpty()) {
239            acc.add(elis);
240          } else {
241            long accFirstTxId = acc.get(0).getFirstTxId();
242            if (accFirstTxId == elis.getFirstTxId()) {
243              acc.add(elis);
244            } else if (accFirstTxId < elis.getFirstTxId()) {
245              streams.add(acc.get(0));
246              acc.clear();
247              acc.add(elis);
248            } else if (accFirstTxId > elis.getFirstTxId()) {
249              throw new RuntimeException("sorted set invariants violated!  " +
250                  "Got stream with first txid " + elis.getFirstTxId() +
251                  ", but the last firstTxId was " + accFirstTxId);
252            }
253          }
254        }
255        if (!acc.isEmpty()) {
256          streams.add(acc.get(0));
257          acc.clear();
258        }
259      }

261      /**
262       * Returns true if there are no journals, all redundant journals are disabled,
263       * or any required journals are disabled.
264       * 
265       * @return True if there no journals, all redundant journals are disabled,
266       * or any required journals are disabled.
267       */
268      public boolean isEmpty() {
269        return !NameNodeResourcePolicy.areResourcesAvailable(journals,
270            minimumRedundantJournals);
271      }

273      /**
274       * Called when some journals experience an error in some operation.
275       */
276      private void disableAndReportErrorOnJournals(List<JournalAndStream> badJournals) {
277        if (badJournals == null || badJournals.isEmpty()) {
278          return; // nothing to do
279        }
280     
281        for (JournalAndStream j : badJournals) {
282          LOG.error("Disabling journal " + j);
283          j.abort();
284          j.setDisabled(true);
285        }
286      }

288      /**
289       * Implementations of this interface encapsulate operations that can be
290       * iteratively applied on all the journals. For example see
291       * {@link JournalSet#mapJournalsAndReportErrors}.
292       */
293      private interface JournalClosure {
294        /**
295         * The operation on JournalAndStream.
296         * @param jas Object on which operations are performed.
297         * @throws IOException
298         */
299        public void apply(JournalAndStream jas) throws IOException;
300      }
301      
302      /**
303       * Apply the given operation across all of the journal managers, disabling
304       * any for which the closure throws an IOException.
305       * @param closure {@link JournalClosure} object encapsulating the operation.
306       * @param status message used for logging errors (e.g. "opening journal")
307       * @throws IOException If the operation fails on all the journals.
308       */
309      private void mapJournalsAndReportErrors(
310          JournalClosure closure, String status) throws IOException{
311    
312        List<JournalAndStream> badJAS = Lists.newLinkedList();
313        for (JournalAndStream jas : journals) {
314          try {
315            closure.apply(jas);
316          } catch (Throwable t) {
317            if (jas.isRequired()) {
318              final String msg = "Error: " + status + " failed for required journal ("
319                + jas + ")";
320              LOG.fatal(msg, t);
321              // If we fail on *any* of the required journals, then we must not
322              // continue on any of the other journals. Abort them to ensure that
323              // retry behavior doesn't allow them to keep going in any way.
324              abortAllJournals();
325              // the current policy is to shutdown the NN on errors to shared edits
326              // dir. There are many code paths to shared edits failures - syncs,
327              // roll of edits etc. All of them go through this common function 
328              // where the isRequired() check is made. Applying exit policy here 
329              // to catch all code paths.
330              terminate(1, msg);
331            } else {
332              LOG.error("Error: " + status + " failed for (journal " + jas + ")", t);
333              badJAS.add(jas);          
334            }
335          }
336        }
337        disableAndReportErrorOnJournals(badJAS);
338        if (!NameNodeResourcePolicy.areResourcesAvailable(journals,
339            minimumRedundantJournals)) {
340          String message = status + " failed for too many journals";
341          LOG.error("Error: " + message);
342          throw new IOException(message);
343        }
344      }
345      
346      /**
347       * Abort all of the underlying streams.
348       */
349      private void abortAllJournals() {
350        for (JournalAndStream jas : journals) {
351          if (jas.isActive()) {
352            jas.abort();
353          }
354        }
355      }

514      void remove(JournalManager j) {
515        JournalAndStream jasToRemove = null;
516        for (JournalAndStream jas: journals) {
517          if (jas.getManager().equals(j)) {
518            jasToRemove = jas;
519            break;
520          }
521        }
522        if (jasToRemove != null) {
523          jasToRemove.abort();
524          journals.remove(jasToRemove);
525        }
526      }

548      /**
549       * Return a manifest of what finalized edit logs are available. All available
550       * edit logs are returned starting from the transaction id passed.
551       * 
552       * @param fromTxId Starting transaction id to read the logs.
553       * @return RemoteEditLogManifest object.
554       */
555      public synchronized RemoteEditLogManifest getEditLogManifest(long fromTxId) {
556        // Collect RemoteEditLogs available from each FileJournalManager
557        List<RemoteEditLog> allLogs = Lists.newArrayList();
558        for (JournalAndStream j : journals) {
559          if (j.getManager() instanceof FileJournalManager) {
560            FileJournalManager fjm = (FileJournalManager)j.getManager();
561            try {
562              allLogs.addAll(fjm.getRemoteEditLogs(fromTxId));
563            } catch (Throwable t) {
564              LOG.warn("Cannot list edit logs in " + fjm, t);
565            }
566          }
567        }
568        
569        // Group logs by their starting txid
570        ImmutableListMultimap<Long, RemoteEditLog> logsByStartTxId =
571          Multimaps.index(allLogs, RemoteEditLog.GET_START_TXID);
572        long curStartTxId = fromTxId;
573    
574        List<RemoteEditLog> logs = Lists.newArrayList();
575        while (true) {
576          ImmutableList<RemoteEditLog> logGroup = logsByStartTxId.get(curStartTxId);
577          if (logGroup.isEmpty()) {
578            // we have a gap in logs - for example because we recovered some old
579            // storage directory with ancient logs. Clear out any logs we've
580            // accumulated so far, and then skip to the next segment of logs
581            // after the gap.
582            SortedSet<Long> startTxIds = Sets.newTreeSet(logsByStartTxId.keySet());
583            startTxIds = startTxIds.tailSet(curStartTxId);
584            if (startTxIds.isEmpty()) {
585              break;
586            } else {
587              if (LOG.isDebugEnabled()) {
588                LOG.debug("Found gap in logs at " + curStartTxId + ": " +
589                    "not returning previous logs in manifest.");
590              }
591              logs.clear();
592              curStartTxId = startTxIds.first();
593              continue;
594            }
595          }
596    
597          // Find the one that extends the farthest forward
598          RemoteEditLog bestLog = Collections.max(logGroup);
599          logs.add(bestLog);
600          // And then start looking from after that point
601          curStartTxId = bestLog.getEndTxId() + 1;
602        }
603        RemoteEditLogManifest ret = new RemoteEditLogManifest(logs);
604        
605        if (LOG.isDebugEnabled()) {
606          LOG.debug("Generated manifest for logs since " + fromTxId + ":"
607              + ret);      
608        }
609        return ret;
610      }

612      /**
613       * Add sync times to the buffer.
614       */
615      String getSyncTimes() {
616        StringBuilder buf = new StringBuilder();
617        for (JournalAndStream jas : journals) {

620            buf.append(" ");
621          }
