060      public static void downloadMostRecentImageToDirectory(String fsName,
061          File dir) throws IOException {
062        String fileId = GetImageServlet.getParamStringForMostRecentImage();
063        getFileClient(fsName, fileId, Lists.newArrayList(dir),
064            null, false);
065      }

067      public static MD5Hash downloadImageToStorage(
068          String fsName, long imageTxId, Storage dstStorage, boolean needDigest)
069          throws IOException {
070        String fileid = GetImageServlet.getParamStringForImage(
071            imageTxId, dstStorage);
072        String fileName = NNStorage.getCheckpointImageFileName(imageTxId);
073        
074        List<File> dstFiles = dstStorage.getFiles(
075            NameNodeDirType.IMAGE, fileName);
076        if (dstFiles.isEmpty()) {
077          throw new IOException("No targets in destination storage!");
078        }
079        
080        MD5Hash hash = getFileClient(fsName, fileid, dstFiles, dstStorage, needDigest);
081        LOG.info("Downloaded file " + dstFiles.get(0).getName() + " size " +
082            dstFiles.get(0).length() + " bytes.");
083        return hash;
084      }
085      
086      static void downloadEditsToStorage(String fsName, RemoteEditLog log,
087          NNStorage dstStorage) throws IOException {
088        assert log.getStartTxId() > 0 && log.getEndTxId() > 0 :
089          "bad log: " + log;
090        String fileid = GetImageServlet.getParamStringForLog(
091            log, dstStorage);
092        String fileName = NNStorage.getFinalizedEditsFileName(
093            log.getStartTxId(), log.getEndTxId());
094    
095        List<File> dstFiles = dstStorage.getFiles(NameNodeDirType.EDITS, fileName);
096        assert !dstFiles.isEmpty() : "No checkpoint targets.";
097        
098        for (File f : dstFiles) {
099          if (f.exists() && f.canRead()) {
100            LOG.info("Skipping download of remote edit log " +
101                log + " since it already is stored locally at " + f);
102            return;
103          } else {
104            LOG.debug("Dest file: " + f);
105          }
106        }
107    
108        getFileClient(fsName, fileid, dstFiles, dstStorage, false);
109        LOG.info("Downloaded file " + dstFiles.get(0).getName() + " size " +
110            dstFiles.get(0).length() + " bytes.");
111      }

113      /**
114       * Requests that the NameNode download an image from this node.
115       *
116       * @param fsName the http address for the remote NN
117       * @param imageListenAddress the host/port where the local node is running an
118       *                           HTTPServer hosting GetImageServlet
119       * @param storage the storage directory to transfer the image from
120       * @param txid the transaction ID of the image to be uploaded
121       */
122      public static void uploadImageFromStorage(String fsName,
123          InetSocketAddress imageListenAddress,
124          Storage storage, long txid) throws IOException {
125        
126        String fileid = GetImageServlet.getParamStringToPutImage(
127            txid, imageListenAddress, storage);
128        // this doesn't directly upload an image, but rather asks the NN
129        // to connect back to the 2NN to download the specified image.
130        try {
131          TransferFsImage.getFileClient(fsName, fileid, null, null, false);
132        } catch (HttpGetFailedException e) {
133          if (e.getResponseCode() == HttpServletResponse.SC_CONFLICT) {
134            // this is OK - this means that a previous attempt to upload
135            // this checkpoint succeeded even though we thought it failed.
136            LOG.info("Image upload with txid " + txid + 
137                " conflicted with a previous image upload to the " +
138                "same NameNode. Continuing...", e);
139            return;
140          } else {
141            throw e;
142          }
143        }
144        LOG.info("Uploaded image with txid " + txid + " to namenode at " +
145                    fsName);
146      }

149      /**
150       * A server-side method to respond to a getfile http request
151       * Copies the contents of the local file into the output stream.
152       */
153      public static void getFileServer(ServletResponse response, File localfile,
154          FileInputStream infile,
155          DataTransferThrottler throttler) 
156        throws IOException {
157        byte buf[] = new byte[HdfsConstants.IO_FILE_BUFFER_SIZE];
158        ServletOutputStream out = null;
159        try {
160          CheckpointFaultInjector.getInstance()
161              .aboutToSendFile(localfile);
162          out = response.getOutputStream();
163    
164          if (CheckpointFaultInjector.getInstance().
165                shouldSendShortFile(localfile)) {
166              // Test sending image shorter than localfile
167              long len = localfile.length();
168              buf = new byte[(int)Math.min(len/2, HdfsConstants.IO_FILE_BUFFER_SIZE)];
169              // This will read at most half of the image
170              // and the rest of the image will be sent over the wire
171              infile.read(buf);
172          }
173          int num = 1;
174          while (num > 0) {
175            num = infile.read(buf);
176            if (num <= 0) {
177              break;
178            }
179            if (CheckpointFaultInjector.getInstance()
180                  .shouldCorruptAByte(localfile)) {
181              // Simulate a corrupted byte on the wire
182              LOG.warn("SIMULATING A CORRUPT BYTE IN IMAGE TRANSFER!");
183              buf[0]++;
184            }
185            
186            out.write(buf, 0, num);
187            if (throttler != null) {
188              throttler.throttle(num);
189            }
190          }
191        } finally {
192          if (out != null) {
193            out.close();
194          }
195        }
196      }

198      /**
199       * Client-side Method to fetch file from a server
200       * Copies the response from the URL to a list of local files.
201       * @param dstStorage if an error occurs writing to one of the files,
202       *                   this storage object will be notified. 
203       * @Return a digest of the received file if getChecksum is true
204       */
205      static MD5Hash getFileClient(String nnHostPort,
206          String queryString, List<File> localPaths,
207          Storage dstStorage, boolean getChecksum) throws IOException {
208        byte[] buf = new byte[HdfsConstants.IO_FILE_BUFFER_SIZE];
209    
210        String str = HttpConfig.getSchemePrefix() + nnHostPort + "/getimage?" +
211            queryString;
212        LOG.info("Opening connection to " + str);
213        //
214        // open connection to remote server
215        //
216        URL url = new URL(str);
217        return doGetUrl(url, localPaths, dstStorage, getChecksum);
218      }

220      public static MD5Hash doGetUrl(URL url, List<File> localPaths,
221          Storage dstStorage, boolean getChecksum) throws IOException {
222        long startTime = Time.monotonicNow();
223    
224        HttpURLConnection connection = (HttpURLConnection)
225          SecurityUtil.openSecureHttpConnection(url);
226    
227        if (connection.getResponseCode() != HttpURLConnection.HTTP_OK) {
228          throw new HttpGetFailedException(
229              "Image transfer servlet at " + url +
230              " failed with status code " + connection.getResponseCode() +
231              "\nResponse message:\n" + connection.getResponseMessage(),
232              connection);
233        }
234        
235        long advertisedSize;
236        String contentLength = connection.getHeaderField(CONTENT_LENGTH);
237        if (contentLength != null) {
238          advertisedSize = Long.parseLong(contentLength);
239        } else {
240          throw new IOException(CONTENT_LENGTH + " header is not provided " +
241                                "by the namenode when trying to fetch " + url);
242        }
243        
244        if (localPaths != null) {
245          String fsImageName = connection.getHeaderField(
246              GetImageServlet.HADOOP_IMAGE_EDITS_HEADER);
247          // If the local paths refer to directories, use the server-provided header
248          // as the filename within that directory
249          List<File> newLocalPaths = new ArrayList<File>();
250          for (File localPath : localPaths) {
251            if (localPath.isDirectory()) {
252              if (fsImageName == null) {
253                throw new IOException("No filename header provided by server");
254              }
255              newLocalPaths.add(new File(localPath, fsImageName));
256            } else {
257              newLocalPaths.add(localPath);
258            }
259          }
260          localPaths = newLocalPaths;
261        }
262        
263        MD5Hash advertisedDigest = parseMD5Header(connection);
264    
265        long received = 0;
266        InputStream stream = connection.getInputStream();
267        MessageDigest digester = null;
268        if (getChecksum) {
269          digester = MD5Hash.getDigester();
270          stream = new DigestInputStream(stream, digester);
271        }
272        boolean finishedReceiving = false;
273    
274        List<FileOutputStream> outputStreams = Lists.newArrayList();
275    
276        try {
277          if (localPaths != null) {
278            for (File f : localPaths) {
279              try {
280                if (f.exists()) {
281                  LOG.warn("Overwriting existing file " + f
282                      + " with file downloaded from " + url);
283                }
284                outputStreams.add(new FileOutputStream(f));
285              } catch (IOException ioe) {
286                LOG.warn("Unable to download file " + f, ioe);
287                // This will be null if we're downloading the fsimage to a file
288                // outside of an NNStorage directory.
289                if (dstStorage != null &&
290                    (dstStorage instanceof StorageErrorReporter)) {
291                  ((StorageErrorReporter)dstStorage).reportErrorOnFile(f);
292                }
293              }
294            }
295            
296            if (outputStreams.isEmpty()) {
297              throw new IOException(
298                  "Unable to download to any storage directory");
299            }
300          }
301          
302          int num = 1;
303          byte[] buf = new byte[HdfsConstants.IO_FILE_BUFFER_SIZE];
304          while (num > 0) {
305            num = stream.read(buf);
306            if (num > 0) {
307              received += num;
308              for (FileOutputStream fos : outputStreams) {
309                fos.write(buf, 0, num);
310              }
311            }
312          }
313          finishedReceiving = true;
314        } finally {
315          stream.close();
316          for (FileOutputStream fos : outputStreams) {
317            fos.getChannel().force(true);
318            fos.close();
319          }
320          if (finishedReceiving && received != advertisedSize) {
321            // only throw this exception if we think we read all of it on our end
322            // -- otherwise a client-side IOException would be masked by this
323            // exception that makes it look like a server-side problem!
324            throw new IOException("File " + url + " received length " + received +
325                                  " is not of the advertised size " +
326                                  advertisedSize);
327          }
328        }
329        double xferSec = Math.max(
330            ((float)(Time.monotonicNow() - startTime)) / 1000.0, 0.001);
331        long xferKb = received / 1024;
332        LOG.info(String.format("Transfer took %.2fs at %.2f KB/s",
333            xferSec, xferKb / xferSec));
334    
335        if (digester != null) {
336          MD5Hash computedDigest = new MD5Hash(digester.digest());
337          
338          if (advertisedDigest != null &&
339              !computedDigest.equals(advertisedDigest)) {
340            throw new IOException("File " + url + " computed digest " +
341                computedDigest + " does not match advertised digest " + 
342                advertisedDigest);
343          }
344          return computedDigest;
345        } else {
346          return null;
347        }    
348      }
349    
350      private static MD5Hash parseMD5Header(HttpURLConnection connection) {
351        String header = connection.getHeaderField(MD5_HEADER);
352        return (header != null) ? new MD5Hash(header) : null;
353      }

355      public static class HttpGetFailedException extends IOException {
356        private static final long serialVersionUID = 1L;
357        private final int responseCode;
358    
359        HttpGetFailedException(String msg, HttpURLConnection connection) throws IOException {
360          super(msg);
361          this.responseCode = connection.getResponseCode();
362        }
363        
364        public int getResponseCode() {
365          return responseCode;
366        }
367      }
