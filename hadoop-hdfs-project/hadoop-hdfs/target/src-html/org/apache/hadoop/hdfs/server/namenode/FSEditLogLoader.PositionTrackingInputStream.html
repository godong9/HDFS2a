081      /**
082       * Load an edit log, and apply the changes to the in-memory structure
083       * This is where we apply edits that we've been writing to disk all
084       * along.
085       */
086      long loadFSEdits(EditLogInputStream edits, long expectedStartingTxId,
087          MetaRecoveryContext recovery) throws IOException {
088        fsNamesys.writeLock();
089        try {
090          long startTime = now();
091          long numEdits = loadEditRecords(edits, false, 
092                                     expectedStartingTxId, recovery);
093          FSImage.LOG.info("Edits file " + edits.getName() 
094              + " of size " + edits.length() + " edits # " + numEdits 
095              + " loaded in " + (now()-startTime)/1000 + " seconds.");
096          return numEdits;
097        } finally {
098          edits.close();
099          fsNamesys.writeUnlock();
100        }
101      }
102    
103      long loadEditRecords(EditLogInputStream in, boolean closeOnExit,
104                          long expectedStartingTxId, MetaRecoveryContext recovery)
105          throws IOException {
106        FSDirectory fsDir = fsNamesys.dir;
107    
108        EnumMap<FSEditLogOpCodes, Holder<Integer>> opCounts =
109          new EnumMap<FSEditLogOpCodes, Holder<Integer>>(FSEditLogOpCodes.class);
110    
111        if (LOG.isTraceEnabled()) {
112          LOG.trace("Acquiring write lock to replay edit log");
113        }
114    
115        fsNamesys.writeLock();
116        fsDir.writeLock();
117    
118        long recentOpcodeOffsets[] = new long[4];
119        Arrays.fill(recentOpcodeOffsets, -1);
120        
121        long expectedTxId = expectedStartingTxId;
122        long numEdits = 0;
123        long lastTxId = in.getLastTxId();
124        long numTxns = (lastTxId - expectedStartingTxId) + 1;
125        long lastLogTime = now();
126    
127        if (LOG.isDebugEnabled()) {
128          LOG.debug("edit log length: " + in.length() + ", start txid: "
129              + expectedStartingTxId + ", last txid: " + lastTxId);
130        }
131        try {
132          while (true) {
133            try {
134              FSEditLogOp op;
135              try {
136                op = in.readOp();
137                if (op == null) {
138                  break;
139                }
140              } catch (Throwable e) {
141                // Handle a problem with our input
142                check203UpgradeFailure(in.getVersion(), e);
143                String errorMessage =
144                  formatEditLogReplayError(in, recentOpcodeOffsets, expectedTxId);
145                FSImage.LOG.error(errorMessage, e);
146                if (recovery == null) {
147                   // We will only try to skip over problematic opcodes when in
148                   // recovery mode.
149                  throw new EditLogInputException(errorMessage, e, numEdits);
150                }
151                MetaRecoveryContext.editLogLoaderPrompt(
152                    "We failed to read txId " + expectedTxId,
153                    recovery, "skipping the bad section in the log");
154                in.resync();
155                continue;
156              }
157              recentOpcodeOffsets[(int)(numEdits % recentOpcodeOffsets.length)] =
158                in.getPosition();
159              if (op.hasTransactionId()) {
160                if (op.getTransactionId() > expectedTxId) { 
161                  MetaRecoveryContext.editLogLoaderPrompt("There appears " +
162                      "to be a gap in the edit log.  We expected txid " +
163                      expectedTxId + ", but got txid " +
164                      op.getTransactionId() + ".", recovery, "ignoring missing " +
165                      " transaction IDs");
166                } else if (op.getTransactionId() < expectedTxId) { 
167                  MetaRecoveryContext.editLogLoaderPrompt("There appears " +
168                      "to be an out-of-order edit in the edit log.  We " +
169                      "expected txid " + expectedTxId + ", but got txid " +
170                      op.getTransactionId() + ".", recovery,
171                      "skipping the out-of-order edit");
172                  continue;
173                }
174              }
175              try {
176                applyEditLogOp(op, fsDir, in.getVersion());
177              } catch (Throwable e) {
178                LOG.error("Encountered exception on operation " + op, e);
179                MetaRecoveryContext.editLogLoaderPrompt("Failed to " +
180                 "apply edit log operation " + op + ": error " +
181                 e.getMessage(), recovery, "applying edits");
182              }
183              // Now that the operation has been successfully decoded and
184              // applied, update our bookkeeping.
185              incrOpCount(op.opCode, opCounts);
186              if (op.hasTransactionId()) {
187                lastAppliedTxId = op.getTransactionId();
188                expectedTxId = lastAppliedTxId + 1;
189              } else {
190                expectedTxId = lastAppliedTxId = expectedStartingTxId;
191              }
192              // log progress
193              if (op.hasTransactionId()) {
194                long now = now();
195                if (now - lastLogTime > REPLAY_TRANSACTION_LOG_INTERVAL) {
196                  int percent = Math.round((float)lastAppliedTxId / numTxns * 100);
197                  LOG.info("replaying edit log: " + lastAppliedTxId + "/" + numTxns
198                      + " transactions completed. (" + percent + "%)");
199                  lastLogTime = now;
200                }
201              }
202              numEdits++;
203            } catch (MetaRecoveryContext.RequestStopException e) {
204              MetaRecoveryContext.LOG.warn("Stopped reading edit log at " +
205                  in.getPosition() + "/"  + in.length());
206              break;
207            }
208          }
209        } finally {
210          if(closeOnExit) {
211            in.close();
212          }
213          fsDir.writeUnlock();
214          fsNamesys.writeUnlock();
215    
216          if (LOG.isTraceEnabled()) {
217            LOG.trace("replaying edit log finished");
218          }
219    
220          if (FSImage.LOG.isDebugEnabled()) {
221            dumpOpCounts(opCounts);
222          }
223        }
224        return numEdits;
225      }

227      @SuppressWarnings("deprecation")
228      private void applyEditLogOp(FSEditLogOp op, FSDirectory fsDir,
229          int logVersion) throws IOException {
230    
231        if (LOG.isTraceEnabled()) {
232          LOG.trace("replaying edit log: " + op);
233        }
234    
235        switch (op.opCode) {
236        case OP_ADD: {
237          AddCloseOp addCloseOp = (AddCloseOp)op;
238          if (FSNamesystem.LOG.isDebugEnabled()) {
239            FSNamesystem.LOG.debug(op.opCode + ": " + addCloseOp.path +
240                " numblocks : " + addCloseOp.blocks.length +
241                " clientHolder " + addCloseOp.clientName +
242                " clientMachine " + addCloseOp.clientMachine);
243          }
244          // There three cases here:
245          // 1. OP_ADD to create a new file
246          // 2. OP_ADD to update file blocks
247          // 3. OP_ADD to open file for append
248    
249          // See if the file already exists (persistBlocks call)
250          INodeFile oldFile = getINodeFile(fsDir, addCloseOp.path);
251          INodeFile newFile = oldFile;
252          if (oldFile == null) { // this is OP_ADD on a new file (case 1)
253            // versions > 0 support per file replication
254            // get name and replication
255            final short replication  = fsNamesys.getBlockManager(
256                ).adjustReplication(addCloseOp.replication);
257            assert addCloseOp.blocks.length == 0;
258    
259            // add to the file tree
260            newFile = (INodeFile)fsDir.unprotectedAddFile(
261                addCloseOp.path, addCloseOp.permissions,
262                replication, addCloseOp.mtime,
263                addCloseOp.atime, addCloseOp.blockSize,
264                true, addCloseOp.clientName, addCloseOp.clientMachine);
265            fsNamesys.leaseManager.addLease(addCloseOp.clientName, addCloseOp.path);
266    
267          } else { // This is OP_ADD on an existing file
268            if (!oldFile.isUnderConstruction()) {
269              // This is case 3: a call to append() on an already-closed file.
270              if (FSNamesystem.LOG.isDebugEnabled()) {
271                FSNamesystem.LOG.debug("Reopening an already-closed file " +
272                    "for append");
273              }
274              fsNamesys.prepareFileForWrite(addCloseOp.path, oldFile,
275                  addCloseOp.clientName, addCloseOp.clientMachine, null,
276                  false);
277              newFile = getINodeFile(fsDir, addCloseOp.path);
278            }
279          }
280          // Fall-through for case 2.
281          // Regardless of whether it's a new file or an updated file,
282          // update the block list.
283          
284          // Update the salient file attributes.
285          newFile.setAccessTime(addCloseOp.atime);
286          newFile.setModificationTimeForce(addCloseOp.mtime);
287          updateBlocks(fsDir, addCloseOp, newFile);
288          break;
289        }
290        case OP_CLOSE: {
291          AddCloseOp addCloseOp = (AddCloseOp)op;
292          
293          if (FSNamesystem.LOG.isDebugEnabled()) {
294            FSNamesystem.LOG.debug(op.opCode + ": " + addCloseOp.path +
295                " numblocks : " + addCloseOp.blocks.length +
296                " clientHolder " + addCloseOp.clientName +
297                " clientMachine " + addCloseOp.clientMachine);
298          }
299    
300          INodeFile oldFile = getINodeFile(fsDir, addCloseOp.path);
301          if (oldFile == null) {
302            throw new IOException("Operation trying to close non-existent file " +
303                addCloseOp.path);
304          }
305          
306          // Update in-memory data structures
307          updateBlocks(fsDir, addCloseOp, oldFile);
308    
309          // Now close the file
310          if (!oldFile.isUnderConstruction() &&
311              logVersion <= LayoutVersion.BUGFIX_HDFS_2991_VERSION) {
312            // There was a bug (HDFS-2991) in hadoop < 0.23.1 where OP_CLOSE
313            // could show up twice in a row. But after that version, this
314            // should be fixed, so we should treat it as an error.
315            throw new IOException(
316                "File is not under construction: " + addCloseOp.path);
317          }
318          // One might expect that you could use removeLease(holder, path) here,
319          // but OP_CLOSE doesn't serialize the holder. So, remove by path.
320          if (oldFile.isUnderConstruction()) {
321            INodeFileUnderConstruction ucFile = (INodeFileUnderConstruction) oldFile;
322            fsNamesys.leaseManager.removeLeaseWithPrefixPath(addCloseOp.path);
323            INodeFile newFile = ucFile.convertToInodeFile();
324            fsDir.replaceNode(addCloseOp.path, ucFile, newFile);
325          }
326          break;
327        }
328        case OP_UPDATE_BLOCKS: {
329          UpdateBlocksOp updateOp = (UpdateBlocksOp)op;
330          if (FSNamesystem.LOG.isDebugEnabled()) {
331            FSNamesystem.LOG.debug(op.opCode + ": " + updateOp.path +
332                " numblocks : " + updateOp.blocks.length);
333          }
334          INodeFile oldFile = getINodeFile(fsDir, updateOp.path);
335          if (oldFile == null) {
336            throw new IOException(
337                "Operation trying to update blocks in non-existent file " +
338                updateOp.path);
339          }
340          
341          // Update in-memory data structures
342          updateBlocks(fsDir, updateOp, oldFile);
343          break;
344        }
345          
346        case OP_SET_REPLICATION: {
347          SetReplicationOp setReplicationOp = (SetReplicationOp)op;
348          short replication = fsNamesys.getBlockManager().adjustReplication(
349              setReplicationOp.replication);
350          fsDir.unprotectedSetReplication(setReplicationOp.path,
351                                          replication, null);
352          break;
353        }
354        case OP_CONCAT_DELETE: {
355          ConcatDeleteOp concatDeleteOp = (ConcatDeleteOp)op;
356          fsDir.unprotectedConcat(concatDeleteOp.trg, concatDeleteOp.srcs,
357              concatDeleteOp.timestamp);
358          break;
359        }
360        case OP_RENAME_OLD: {
361          RenameOldOp renameOp = (RenameOldOp)op;
362          HdfsFileStatus dinfo = fsDir.getFileInfo(renameOp.dst, false);
363          fsDir.unprotectedRenameTo(renameOp.src, renameOp.dst,
364                                    renameOp.timestamp);
365          fsNamesys.unprotectedChangeLease(renameOp.src, renameOp.dst, dinfo);
366          break;
367        }
368        case OP_DELETE: {
369          DeleteOp deleteOp = (DeleteOp)op;
370          fsDir.unprotectedDelete(deleteOp.path, deleteOp.timestamp);
371          break;
372        }
373        case OP_MKDIR: {
374          MkdirOp mkdirOp = (MkdirOp)op;
375          fsDir.unprotectedMkdir(mkdirOp.path, mkdirOp.permissions,
376                                 mkdirOp.timestamp);
377          break;
378        }
379        case OP_SET_GENSTAMP: {
380          SetGenstampOp setGenstampOp = (SetGenstampOp)op;
381          fsNamesys.setGenerationStamp(setGenstampOp.genStamp);
382          break;
383        }
384        case OP_SET_PERMISSIONS: {
385          SetPermissionsOp setPermissionsOp = (SetPermissionsOp)op;
386          fsDir.unprotectedSetPermission(setPermissionsOp.src,
387                                         setPermissionsOp.permissions);
388          break;
389        }
390        case OP_SET_OWNER: {
391          SetOwnerOp setOwnerOp = (SetOwnerOp)op;
392          fsDir.unprotectedSetOwner(setOwnerOp.src, setOwnerOp.username,
393                                    setOwnerOp.groupname);
394          break;
395        }
396        case OP_SET_NS_QUOTA: {
397          SetNSQuotaOp setNSQuotaOp = (SetNSQuotaOp)op;
398          fsDir.unprotectedSetQuota(setNSQuotaOp.src,
399                                    setNSQuotaOp.nsQuota,
400                                    HdfsConstants.QUOTA_DONT_SET);
401          break;
402        }
403        case OP_CLEAR_NS_QUOTA: {
404          ClearNSQuotaOp clearNSQuotaOp = (ClearNSQuotaOp)op;
405          fsDir.unprotectedSetQuota(clearNSQuotaOp.src,
406                                    HdfsConstants.QUOTA_RESET,
407                                    HdfsConstants.QUOTA_DONT_SET);
408          break;
409        }
410    
411        case OP_SET_QUOTA:
412          SetQuotaOp setQuotaOp = (SetQuotaOp)op;
413          fsDir.unprotectedSetQuota(setQuotaOp.src,
414                                    setQuotaOp.nsQuota,
415                                    setQuotaOp.dsQuota);
416          break;
417    
418        case OP_TIMES: {
419          TimesOp timesOp = (TimesOp)op;
420    
421          fsDir.unprotectedSetTimes(timesOp.path,
422                                    timesOp.mtime,
423                                    timesOp.atime, true);
424          break;
425        }
426        case OP_SYMLINK: {
427          SymlinkOp symlinkOp = (SymlinkOp)op;
428          fsDir.unprotectedSymlink(symlinkOp.path, symlinkOp.value,
429                                   symlinkOp.mtime, symlinkOp.atime,
430                                   symlinkOp.permissionStatus);
431          break;
432        }
433        case OP_RENAME: {
434          RenameOp renameOp = (RenameOp)op;
435    
436          HdfsFileStatus dinfo = fsDir.getFileInfo(renameOp.dst, false);
437          fsDir.unprotectedRenameTo(renameOp.src, renameOp.dst,
438                                    renameOp.timestamp, renameOp.options);
439          fsNamesys.unprotectedChangeLease(renameOp.src, renameOp.dst, dinfo);
440          break;
441        }
442        case OP_GET_DELEGATION_TOKEN: {
443          GetDelegationTokenOp getDelegationTokenOp
444            = (GetDelegationTokenOp)op;
445    
446          fsNamesys.getDelegationTokenSecretManager()
447            .addPersistedDelegationToken(getDelegationTokenOp.token,
448                                         getDelegationTokenOp.expiryTime);
449          break;
450        }
451        case OP_RENEW_DELEGATION_TOKEN: {
452          RenewDelegationTokenOp renewDelegationTokenOp
453            = (RenewDelegationTokenOp)op;
454          fsNamesys.getDelegationTokenSecretManager()
455            .updatePersistedTokenRenewal(renewDelegationTokenOp.token,
456                                         renewDelegationTokenOp.expiryTime);
457          break;
458        }
459        case OP_CANCEL_DELEGATION_TOKEN: {
460          CancelDelegationTokenOp cancelDelegationTokenOp
461            = (CancelDelegationTokenOp)op;
462          fsNamesys.getDelegationTokenSecretManager()
463              .updatePersistedTokenCancellation(
464                  cancelDelegationTokenOp.token);
465          break;
466        }
467        case OP_UPDATE_MASTER_KEY: {
468          UpdateMasterKeyOp updateMasterKeyOp = (UpdateMasterKeyOp)op;
469          fsNamesys.getDelegationTokenSecretManager()
470            .updatePersistedMasterKey(updateMasterKeyOp.key);
471          break;
472        }
473        case OP_REASSIGN_LEASE: {
474          ReassignLeaseOp reassignLeaseOp = (ReassignLeaseOp)op;
475    
476          Lease lease = fsNamesys.leaseManager.getLease(
477              reassignLeaseOp.leaseHolder);
478          INodeFileUnderConstruction pendingFile =
479              (INodeFileUnderConstruction) fsDir.getFileINode(
480                  reassignLeaseOp.path);
481          fsNamesys.reassignLeaseInternal(lease,
482              reassignLeaseOp.path, reassignLeaseOp.newHolder, pendingFile);
483          break;
484        }
485        case OP_START_LOG_SEGMENT:
486        case OP_END_LOG_SEGMENT: {
487          // no data in here currently.
488          break;
489        }
490        default:
491          throw new IOException("Invalid operation read " + op.opCode);
492        }
493      }

495      private static String formatEditLogReplayError(EditLogInputStream in,
496          long recentOpcodeOffsets[], long txid) {
497        StringBuilder sb = new StringBuilder();
498        sb.append("Error replaying edit log at offset " + in.getPosition());
499        sb.append(".  Expected transaction ID was ").append(txid);
500        if (recentOpcodeOffsets[0] != -1) {
501          Arrays.sort(recentOpcodeOffsets);
502          sb.append("\nRecent opcode offsets:");
503          for (long offset : recentOpcodeOffsets) {
504            if (offset != -1) {
505              sb.append(' ').append(offset);
506            }
507          }
508        }
509        return sb.toString();
510      }

512      private static INodeFile getINodeFile(FSDirectory fsDir, String path)
513          throws IOException {
514        INode inode = fsDir.getINode(path);
515        if (inode != null) {
516          if (!(inode instanceof INodeFile)) {
517            throw new IOException("Operation trying to get non-file " + path);
518          }
519        }
520        return (INodeFile)inode;
521      }

523      /**
524       * Update in-memory data structures with new block information.
525       * @throws IOException
526       */
527      private void updateBlocks(FSDirectory fsDir, BlockListUpdatingOp op,
528          INodeFile file) throws IOException {
529        // Update its block list
530        BlockInfo[] oldBlocks = file.getBlocks();
531        Block[] newBlocks = op.getBlocks();
532        String path = op.getPath();
533        
534        // Are we only updating the last block's gen stamp.
535        boolean isGenStampUpdate = oldBlocks.length == newBlocks.length;
536        
537        // First, update blocks in common
538        for (int i = 0; i < oldBlocks.length && i < newBlocks.length; i++) {
539          BlockInfo oldBlock = oldBlocks[i];
540          Block newBlock = newBlocks[i];
541          
542          boolean isLastBlock = i == newBlocks.length - 1;
543          if (oldBlock.getBlockId() != newBlock.getBlockId() ||
544              (oldBlock.getGenerationStamp() != newBlock.getGenerationStamp() && 
545                  !(isGenStampUpdate && isLastBlock))) {
546            throw new IOException("Mismatched block IDs or generation stamps, " +
547                "attempting to replace block " + oldBlock + " with " + newBlock +
548                " as block # " + i + "/" + newBlocks.length + " of " +
549                path);
550          }
551          
552          oldBlock.setNumBytes(newBlock.getNumBytes());
553          boolean changeMade =
554            oldBlock.getGenerationStamp() != newBlock.getGenerationStamp();
555          oldBlock.setGenerationStamp(newBlock.getGenerationStamp());
556          
557          if (oldBlock instanceof BlockInfoUnderConstruction &&
558              (!isLastBlock || op.shouldCompleteLastBlock())) {
559            changeMade = true;
560            fsNamesys.getBlockManager().forceCompleteBlock(
561                (INodeFileUnderConstruction)file,
562                (BlockInfoUnderConstruction)oldBlock);
563          }
564          if (changeMade) {
565            // The state or gen-stamp of the block has changed. So, we may be
566            // able to process some messages from datanodes that we previously
567            // were unable to process.
568            fsNamesys.getBlockManager().processQueuedMessagesForBlock(newBlock);
569          }
570        }
571        
572        if (newBlocks.length < oldBlocks.length) {
573          // We're removing a block from the file, e.g. abandonBlock(...)
574          if (!file.isUnderConstruction()) {
575            throw new IOException("Trying to remove a block from file " +
576                path + " which is not under construction.");
577          }
578          if (newBlocks.length != oldBlocks.length - 1) {
579            throw new IOException("Trying to remove more than one block from file "
580                + path);
581          }
582          fsDir.unprotectedRemoveBlock(path,
583              (INodeFileUnderConstruction)file, oldBlocks[oldBlocks.length - 1]);
584        } else if (newBlocks.length > oldBlocks.length) {
585          // We're adding blocks
586          for (int i = oldBlocks.length; i < newBlocks.length; i++) {
587            Block newBlock = newBlocks[i];
588            BlockInfo newBI;
589            if (!op.shouldCompleteLastBlock()) {
590              // TODO: shouldn't this only be true for the last block?
591              // what about an old-version fsync() where fsync isn't called
592              // until several blocks in?
593              newBI = new BlockInfoUnderConstruction(
594                  newBlock, file.getReplication());
595            } else {
596              // OP_CLOSE should add finalized blocks. This code path
597              // is only executed when loading edits written by prior
598              // versions of Hadoop. Current versions always log
599              // OP_ADD operations as each block is allocated.
600              newBI = new BlockInfo(newBlock, file.getReplication());
601            }
602            fsNamesys.getBlockManager().addBlockCollection(newBI, file);
603            file.addBlock(newBI);
604            fsNamesys.getBlockManager().processQueuedMessagesForBlock(newBlock);
605          }
606        }
607      }

618          EnumMap<FSEditLogOpCodes, Holder<Integer>> opCounts) {
619        Holder<Integer> holder = opCounts.get(opCode);
620        if (holder == null) {

622          opCounts.put(opCode, holder);

628      /**
629       * Throw appropriate exception during upgrade from 203, when editlog loading
630       * could fail due to opcode conflicts.
631       */
632      private void check203UpgradeFailure(int logVersion, Throwable e)
633          throws IOException {
634        // 0.20.203 version version has conflicting opcodes with the later releases.
635        // The editlog must be emptied by restarting the namenode, before proceeding
636        // with the upgrade.
637        if (Storage.is203LayoutVersion(logVersion)
638            && logVersion != HdfsConstants.LAYOUT_VERSION) {
639          String msg = "During upgrade failed to load the editlog version "
640              + logVersion + " from release 0.20.203. Please go back to the old "
641              + " release and restart the namenode. This empties the editlog "
642              + " and saves the namespace. Resume the upgrade after this step.";
643          throw new IOException(msg, e);
644        }
645      }

647      /**
648       * Find the last valid transaction ID in the stream.
649       * If there are invalid or corrupt transactions in the middle of the stream,
650       * validateEditLog will skip over them.
651       * This reads through the stream but does not close it.
652       *
653       * @throws IOException if the stream cannot be read due to an IO error (eg
654       *                     if the log does not exist)
655       */
656      static EditLogValidation validateEditLog(EditLogInputStream in) {
657        long lastPos = 0;
658        long lastTxId = HdfsConstants.INVALID_TXID;
659        long numValid = 0;
660        FSEditLogOp op = null;
661        while (true) {
662          lastPos = in.getPosition();
663          try {
664            if ((op = in.readOp()) == null) {
665              break;
666            }
667          } catch (Throwable t) {
668            FSImage.LOG.warn("Caught exception after reading " + numValid +
669                " ops from " + in + " while determining its valid length." +
670                "Position was " + lastPos, t);
671            break;
672          }
673          if (lastTxId == HdfsConstants.INVALID_TXID
674              || op.getTransactionId() > lastTxId) {
675            lastTxId = op.getTransactionId();
676          }
677          numValid++;
678        }
679        return new EditLogValidation(lastPos, lastTxId, false);
680      }

682      static class EditLogValidation {
683        private final long validLength;
684        private final long endTxId;
685        private final boolean hasCorruptHeader;
686    
687        EditLogValidation(long validLength, long endTxId,
688            boolean hasCorruptHeader) {
689          this.validLength = validLength;
690          this.endTxId = endTxId;
691          this.hasCorruptHeader = hasCorruptHeader;
692        }
693    
694        long getValidLength() { return validLength; }
695    
696        long getEndTxId() { return endTxId; }
697    
698        boolean hasCorruptHeader() { return hasCorruptHeader; }
699      }

701      /**
702       * Stream wrapper that keeps track of the current stream position.
703       * 
704       * This stream also allows us to set a limit on how many bytes we can read
705       * without getting an exception.
706       */
707      public static class PositionTrackingInputStream extends FilterInputStream
708          implements StreamLimiter {
709        private long curPos = 0;
710        private long markPos = -1;
711        private long limitPos = Long.MAX_VALUE;
712    
713        public PositionTrackingInputStream(InputStream is) {
714          super(is);
715        }
716    
717        private void checkLimit(long amt) throws IOException {
718          long extra = (curPos + amt) - limitPos;
719          if (extra > 0) {
720            throw new IOException("Tried to read " + amt + " byte(s) past " +
721                "the limit at offset " + limitPos);
722          }
723        }
724        
725        @Override
726        public int read() throws IOException {
727          checkLimit(1);
728          int ret = super.read();
729          if (ret != -1) curPos++;
730          return ret;
731        }
732    
733        @Override
734        public int read(byte[] data) throws IOException {
735          checkLimit(data.length);
736          int ret = super.read(data);
737          if (ret > 0) curPos += ret;
738          return ret;
739        }
740    
741        @Override
742        public int read(byte[] data, int offset, int length) throws IOException {
743          checkLimit(length);
744          int ret = super.read(data, offset, length);
745          if (ret > 0) curPos += ret;
746          return ret;
747        }
748    
749        @Override
750        public void setLimit(long limit) {
751          limitPos = curPos + limit;
752        }
753    
754        @Override
755        public void clearLimit() {
756          limitPos = Long.MAX_VALUE;
757        }
758    
759        @Override
760        public void mark(int limit) {
761          super.mark(limit);
762          markPos = curPos;
763        }
764    
765        @Override
766        public void reset() throws IOException {
767          if (markPos == -1) {
768            throw new IOException("Not marked!");
769          }
770          super.reset();
771          curPos = markPos;
772          markPos = -1;
773        }
774    
775        public long getPos() {
776          return curPos;
777        }
778        
779        @Override
780        public long skip(long amt) throws IOException {
781          long extra = (curPos + amt) - limitPos;
782          if (extra > 0) {
783            throw new IOException("Tried to skip " + extra + " bytes past " +
784                "the limit at offset " + limitPos);
785          }
786          long ret = super.skip(amt);
787          curPos += ret;
788          return ret;
789        }
790      }
791    
792      public long getLastAppliedTxId() {
793        return lastAppliedTxId;
794      }
